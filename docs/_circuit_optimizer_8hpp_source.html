35 private:
36  // Remove adjacent NOT (X) gates:
37  // If it encounters a X gate, it will push that gate to the right
38  // to find a matching X gate hence can cancel both of them.
39  // We can push through the X gate if the qubit wire is the *target* of a CNOT gate.
40  // e.g. X(q[1]); CX(q[0], q[1]); <=> CX(q[0], q[1]); X(q[1]);
41  // If no cancellation can be realized, the move is abandoned, i.e. no change to the original circuit.
42  // - io_program: the composite instruction that we will optimize.
43  // - Return: true if it can cancel at least one X gate; false otherwise.
44  bool tryPermuteAndCancelXGate(std::shared_ptr<CompositeInstruction>& io_program);
45 
46  // Reduce the Hadamard gates in the circuit as much as possiple.
47  // Rationale: Hadamard gates tend to block circuit optimization, i.e. they act as barriers.
48  // Hence, reducing their count (even if the total gate count is the same) could help downstream optimization.
49  // There are 5 circuit patterns of this reduction, please refer to Figure 4 of https://arxiv.org/pdf/1710.07345.pdf for details.
50  // Notes: The phase gate is equivalent to Rz(pi/2).
51  bool tryReduceHadamardGates(std::shared_ptr<CompositeInstruction>& io_program);
52 
53  // Identify and merge Rz rotation gates using the *phase polynomials* representation.
54  // This implements routine #4 in https://arxiv.org/pdf/1710.07345.pdf
55  bool tryRotationMergingUsingPhasePolynomials(std::shared_ptr<CompositeInstruction>& io_program);
56 };
