24 // Util func to get a double parameter from an InstructionParameter variant.
25 // Note: most of the cases, we have double-type parameters (e.g. rotation
26 // angles). This also handles int->double and string->double conversion if
27 // necessary.
28 static double
29 InstructionParameterToDouble(const xacc::InstructionParameter &in_parameter) {
30  if (in_parameter.which() == 0) {
31  // Convert int to double
32  return static_cast<double>(in_parameter.as<int>());
33  } else if (in_parameter.which() == 1) {
34  return in_parameter.as<double>();
35  } else {
36  // Check if this string parameter can be converted to a double
37  const auto isNumber = [](const std::string &in_string,
38  double &out_double) -> bool {
39  char *end = 0;
40  double val = strtod(in_string.c_str(), &end);
41  // This whole string must be a valid double number
42  const bool isConversionOkay =
43  (end != in_string.c_str()) && (*end == '\0') &&
44  (val != std::numeric_limits<double>::infinity());
45  if (isConversionOkay) {
46  out_double = val;
47  }
48 
49  return isConversionOkay;
50  };
51 
52  const std::string paramString = in_parameter.toString();
53  double paramDouble = 0.0;
54  if (isNumber(paramString, paramDouble)) {
55  return paramDouble;
56  }
57  }
58 
59  // Cannot find a way to convert, returns 0.0
60  return 0.0;
61 }

81 // The Instruction interface exposes an API for describing a general
82 // post-Moore's law low-level assembly instruction. Each Instruction
83 // exposes a unique string name, the vector of bit indices that it operates on,
84 // and a vector of InstructionParameters describing any runtime
85 // parameters (like rotation angle for a gate-model quantum computing
86 // instruction) that it requires. Clients can get the name of the
87 // Instruction, get and set the bits it operates on, and get and set
88 // any of the InstructionParameters it exposes.
89 //
90 // Instructions can be enabled or disabled, and expose whether
91 // it is composite or not (see CompositeInstruction).
92 //
93 // Instructions are Cloneable, which is required for use with the
94 // xacc service registry. Therefore, all Instructions are meant to
95 // be provided services within the XACC framework, making it easy
96 // to inject new instructions to an existing framework install.
97 //
98 // Instruction can be visited, enabling type-specific information
99 // at runtime to custom InstructionVisitors.
