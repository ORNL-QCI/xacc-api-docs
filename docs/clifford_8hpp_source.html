200  // initial state = all zeros
201  // add destabilizers
202  #pragma omp parallel for if (num_qubits_ > omp_threshold_ && omp_threads_ > 1) num_threads(omp_threads_)
203  for (int64_t i = 0; i < static_cast<int64_t>(nq); i++) {
204  Pauli::Pauli P(nq);
205  P.X.setValue(1, i);
206  table_.push_back(P);

296 bool Clifford::is_deterministic_outcome(const uint64_t& qubit) const {
297  // Clifford state measurements only have three probabilities:
298  // (p0, p1) = (0.5, 0.5), (1, 0), or (0, 1)
299  // The random case happens if there is a row anti-commuting with Z[qubit]
300  return !z_anticommuting(qubit).first;
301 }

303 bool Clifford::measure_and_update(const uint64_t qubit, const uint64_t randint) {
304  // Clifford state measurements only have three probabilities:
305  // (p0, p1) = (0.5, 0.5), (1, 0), or (0, 1)
306  // The random case happens if there is a row anti-commuting with Z[qubit]
307  auto anticom = z_anticommuting(qubit);
308  if (anticom.first) {
309  bool outcome = (randint == 1);
310  auto row = anticom.second;
311  for (uint64_t i = 0; i < 2 * num_qubits_; i++) {
312  // the last condition is not in the AG paper but we seem to need it
313  if ((table_[i].X[qubit]) && (i != row) && (i != (row - num_qubits_))) {
314  rowsum_helper(table_[row], phases_[row], table_[i], phases_[i]);
315  }
316  }
317  // Update state
318  table_[row - num_qubits_].X = table_[row].X;
319  table_[row - num_qubits_].Z = table_[row].Z;
320  phases_[row - num_qubits_] = phases_[row];
321  table_[row].X.makeZero();
322  table_[row].Z.makeZero();
323  table_[row].Z.setValue(1, qubit);
324  phases_[row] = outcome;
325  return outcome;
326  } else {
327  // Deterministic outcome
328  Pauli::Pauli accum(num_qubits_);
329  phase_t outcome = 0;
330  for (uint64_t i = 0; i < num_qubits_; i++) {
331  if (table_[i].X[qubit]) {
332  rowsum_helper(table_[i + num_qubits_], phases_[i + num_qubits_],
333  accum, outcome);
334  }
335  }
336  return outcome;
337  }

340 int64_t Clifford::expectation_value(const std::vector<uint64_t>& qubits) {
341  // Check if there is a row that anti-commutes with an odd number of qubits
342  // If so expectation value is 0
343  for (uint64_t p = num_qubits_; p < 2 * num_qubits_; p++) {
344  uint64_t num_of_x = 0;
345  for (auto qubit : qubits) {
346  if (table_[p].X[qubit])
347  num_of_x ++;
348  }
349  if(num_of_x % 2 == 1)
350  return 0;
351  }
352 
353  // Otherwise the expectation value is +1 or -1
354  uint64_t sum_of_outcomes = 0;
355  for (auto qubit : qubits) {
356  Pauli::Pauli accum(num_qubits_);
357  phase_t outcome = 0;
358  for (uint64_t i = 0; i < num_qubits_; i++) {
359  if (table_[i].X[qubit]) {
360  rowsum_helper(table_[i + num_qubits_], phases_[i + num_qubits_],
361  accum, outcome);
362  }
363  }
364  sum_of_outcomes += outcome;
365  }
366 
367  return (sum_of_outcomes % 2 == 0) ? 1 : -1;
368 }

370 void Clifford::rowsum_helper(const Pauli::Pauli &row, const phase_t row_phase,
371  Pauli::Pauli &accum, phase_t &accum_phase) const {
372  int8_t newr = ((2 * row_phase + 2 * accum_phase) +
373  Pauli::Pauli::phase_exponent(row, accum)) % 4;
374  // Since we are only using +1 and -1 phases in our Clifford phases
375  // the exponent must be 0 (for +1) or 2 (for -1)
376  if ((newr != 0) && (newr != 2)) {
377  throw std::runtime_error("Clifford: rowsum error");
378  }
379  accum_phase = (newr == 2);
380  accum.X += row.X;
381  accum.Z += row.Z;
382 }
