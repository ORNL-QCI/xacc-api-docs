240 template <typename T> std::vector<T> linspace(T a, T b, size_t N) {
241  T h = (b - a) / static_cast<T>(N - 1);
242  std::vector<T> xs(N);
243  typename std::vector<T>::iterator x;
244  T val;
245  for (x = xs.begin(), val = a; x != xs.end(); ++x, val += h)
246  *x = val;
247  return xs;
248 }
