35 /**************************************************************************
36  * Config settings:
37  * - "optimize_ideal_threshold" (int): Qubit threshold for running circuit
38  * optimizations passes for an ideal circuit [Default: 0].
39  * - "optimize_noise_threshold" (int): Qubit threshold for running circuit
40  * optimizations passes for a noisy circuit [Default: 12].
41  *
42  * From Statevector::State class
43  *
44  * - "initial_statevector" (json complex vector): Use a custom initial
45  * statevector for the simulation [Default: null].
46  * - "zero_threshold" (double): Threshold for truncating small values to
47  * zero in result data [Default: 1e-10]
48  * - "statevector_parallel_threshold" (int): Threshold that number of qubits
49  * must be greater than to enable OpenMP parallelization at State
50  * level [Default: 13]
51  * - "statevector_sample_measure_opt" (int): Threshold that number of qubits
52  * must be greater than to enable indexing optimization during
53  * measure sampling [Default: 10]
54  *
55  * From ExtendedStabilizer::State class
56  * - "extended_stabilizer_approximation_error" (double): Set the error in the
57  * approximation for the ch method. A smaller error needs more
58  * memory and computational time. [Default: 0.05]
59  *
60  * - "extended_stabilizer_disable_measurement_opt" (bool): Force the simulator
61  *to re-run the monte-carlo step for every measurement. Enabling this will
62  *improve the sampling accuracy if the output distribution is strongly peaked,
63  *but requires more computational time. [Default: True]
64  *
65  * - "extended_stabilizer_mixing_time" (int): Set how long the monte-carlo
66  *method runs before performing measurements. If the output distribution is
67  *strongly peaked, this can be decreased alongside setting
68  *extended_stabilizer_disable_measurement_opt to True. [Default: 5000]
69  *
70  * - "extended_stabilizer_norm_estimation_samples" (int): Number of samples used
71  *to compute the correct normalisation for a statevector snapshot. [Default:
72  *100]
73  *
74  * - "extended_stabilizer_parallel_threshold" (int): Set the minimum size of the
75  *ch decomposition before we enable OpenMP parallelisation. If parallel circuit
76  *or shot execution is enabled this will only use unallocated CPU cores up to
77  *max_parallel_threads. [Default: 100]
78  *
79  * From BaseController Class
80  *
81  * - "noise_model" (json): A noise model to use for simulation [Default: null]
82  * - "max_parallel_threads" (int): Set the maximum OpenMP threads that may
83  * be used across all levels of parallelization. Set to 0 for maximum
84  * available. [Default : 0]
85  * - "max_parallel_experiments" (int): Set number of circuits that may be
86  * executed in parallel. Set to 0 to use the number of max parallel
87  * threads [Default: 1]
88  * - "max_parallel_shots" (int): Set number of shots that maybe be executed
89  * in parallel for each circuit. Sset to 0 to use the number of max
90  * parallel threads [Default: 1].
91  * - "counts" (bool): Return counts objecy in circuit data [Default: True]
92  * - "snapshots" (bool): Return snapshots object in circuit data [Default: True]
93  * - "memory" (bool): Return memory array in circuit data [Default: False]
94  * - "register" (bool): Return register array in circuit data [Default: False]
95  * - "max_memory_mb" (int): Memory in MB available to the state class.
96  * If specified, is divided by the number of parallel shots/experiments.
97  * [Default: 0]
98  *
99  * From Transpile:Fusion Class
100  * - fusion_enable (bool): Enable fusion optimization in circuit optimization
101  * passes [Default: True]
102  * - fusion_verbose (bool): Output gates generated in fusion optimization
103  * into metadata [Default: False]
104  * - fusion_max_qubit (int): Maximum number of qubits for a operation generated
105  * in a fusion optimization [Default: 5]
106  * - fusion_threshold (int): Threshold that number of qubits must be greater
107  * than or equal to enable fusion optimization [Default: 20]
108  *
109  **************************************************************************/

285 void QasmController::set_config(const json_t &config) {
286  // Set base controller config
287  Base::Controller::set_config(config);
288 
289  // Override automatic simulation method with a fixed method
290  std::string method;
291  if (JSON::get_value(method, "method", config)) {
292  if (method == "statevector" || method == "statevector_cpu") {
293  simulation_method_ = Method::statevector;
294  } else if (method == "statevector_gpu") {
295  simulation_method_ = Method::statevector_thrust_gpu;
296  } else if (method == "statevector_thrust") {
297  simulation_method_ = Method::statevector_thrust_cpu;
298  } else if (method == "density_matrix" || method == "density_matrix_cpu") {
299  simulation_method_ = Method::density_matrix;
300  } else if (method == "density_matrix_gpu") {
301  simulation_method_ = Method::density_matrix_thrust_gpu;
302  } else if (method == "density_matrix_thrust") {
303  simulation_method_ = Method::density_matrix_thrust_cpu;
304  } else if (method == "stabilizer") {
305  simulation_method_ = Method::stabilizer;
306  } else if (method == "extended_stabilizer") {
307  simulation_method_ = Method::extended_stabilizer;
308  } else if (method == "matrix_product_state") {
309  simulation_method_ = Method::matrix_product_state;
310  } else if (method != "automatic") {
311  throw std::runtime_error(
312  std::string("QasmController: Invalid simulation method (") + method +
313  std::string(")."));
314  }
315  }
316 
317  std::string precision;
318  if (JSON::get_value(precision, "precision", config)) {
319  if (precision == "double") {
320  simulation_precision_ = Precision::double_precision;
321  } else if (precision == "single") {
322  simulation_precision_ = Precision::single_precision;
323  }
324  }
325 
326  // Check for extended stabilizer measure sampling
327  JSON::get_value(extended_stabilizer_measure_sampling_,
328  "extended_stabilizer_measure_sampling", config);
329 
330  // DEPRECATED: Add custom initial state
331  if (JSON::get_value(initial_statevector_, "initial_statevector", config)) {
332  // Raise error if method is set to stabilizer or ch
333  if (simulation_method_ == Method::stabilizer) {
334  throw std::runtime_error(
335  std::string("QasmController: Using an initial statevector") +
336  std::string(" is not valid with stabilizer simulation method.") +
337  method);
338  } else if (simulation_method_ == Method::extended_stabilizer) {
339  throw std::runtime_error(
340  std::string("QasmController: Using an initial statevector") +
341  std::string(" is not valid with the CH simulation method.") + method);
342  }
343  // Override simulator method to statevector
344  simulation_method_ = Method::statevector;
345  // Check initial state is normalized
346  if (!Utils::is_unit_vector(initial_statevector_, validation_threshold_)) {
347  throw std::runtime_error(
348  "QasmController: initial_statevector is not a unit vector");
349  }
350  }
351 }

363 ExperimentData QasmController::run_circuit(const Circuit &circ,
364  const Noise::NoiseModel &noise,
365  const json_t &config, uint_t shots,
366  uint_t rng_seed) const {
367  // Validate circuit for simulation method
368  switch (simulation_method(circ, noise, true)) {
369  case Method::statevector:
370  if (simulation_precision_ == Precision::double_precision) {
371  // Double-precision Statevector simulation
372  return run_circuit_helper<Statevector::State<QV::QubitVector<double>>>(
373  circ, noise, config, shots, rng_seed, initial_statevector_,
374  Method::statevector);
375  } else {
376  // Single-precision Statevector simulation
377  return run_circuit_helper<Statevector::State<QV::QubitVector<float>>>(
378  circ, noise, config, shots, rng_seed, initial_statevector_,
379  Method::statevector);
380  }
381  case Method::statevector_thrust_gpu:
382 #ifndef AER_THRUST_CUDA
383  throw std::runtime_error(
384  "QasmController: method statevector_gpu is not supported on this "
385  "system");
386 #else
387  if (simulation_precision_ == Precision::double_precision) {
388  // Double-precision Statevector simulation
389  return run_circuit_helper<
390  Statevector::State<QV::QubitVectorThrust<double>>>(
391  circ, noise, config, shots, rng_seed, initial_statevector_,
392  Method::statevector_thrust_gpu);
393  } else {
394  // Single-precision Statevector simulation
395  return run_circuit_helper<
396  Statevector::State<QV::QubitVectorThrust<float>>>(
397  circ, noise, config, shots, rng_seed, initial_statevector_,
398  Method::statevector_thrust_gpu);
399  }
400 #endif
401  case Method::statevector_thrust_cpu:
402 #ifndef AER_THRUST_CPU
403  throw std::runtime_error(
404  "QasmController: method statevector_thrust is not supported on this "
405  "system");
406 #else
407  if (simulation_precision_ == Precision::double_precision) {
408  // Double-precision Statevector simulation
409  return run_circuit_helper<
410  Statevector::State<QV::QubitVectorThrust<double>>>(
411  circ, noise, config, shots, rng_seed, initial_statevector_,
412  Method::statevector_thrust_cpu);
413  } else {
414  // Single-precision Statevector simulation
415  return run_circuit_helper<
416  Statevector::State<QV::QubitVectorThrust<float>>>(
417  circ, noise, config, shots, rng_seed, initial_statevector_,
418  Method::statevector_thrust_cpu);
419  }
420 #endif
421  case Method::density_matrix:
422  if (simulation_precision_ == Precision::double_precision) {
423  // Double-precision density matrix simulation
424  return run_circuit_helper<
425  DensityMatrix::State<QV::DensityMatrix<double>>>(
426  circ, noise, config, shots, rng_seed, cvector_t(),
427  Method::density_matrix);
428  } else {
429  // Single-precision density matrix simulation
430  return run_circuit_helper<
431  DensityMatrix::State<QV::DensityMatrix<float>>>(
432  circ, noise, config, shots, rng_seed, cvector_t(),
433  Method::density_matrix);
434  }
435  case Method::density_matrix_thrust_gpu:
436 #ifndef AER_THRUST_CUDA
437  throw std::runtime_error(
438  "QasmController: method density_matrix_gpu is not supported on this "
439  "system");
440 #else
441  if (simulation_precision_ == Precision::double_precision) {
442  // Double-precision density matrix simulation
443  return run_circuit_helper<
444  DensityMatrix::State<QV::DensityMatrixThrust<double>>>(
445  circ, noise, config, shots, rng_seed, cvector_t(),
446  Method::density_matrix_thrust_gpu);
447  } else {
448  // Single-precision density matrix simulation
449  return run_circuit_helper<
450  DensityMatrix::State<QV::DensityMatrixThrust<float>>>(
451  circ, noise, config, shots, rng_seed, cvector_t(),
452  Method::density_matrix_thrust_gpu);
453  }
454 #endif
455  case Method::density_matrix_thrust_cpu:
456 #ifndef AER_THRUST_CPU
457  throw std::runtime_error(
458  "QasmController: method density_matrix_thrust is not supported on this "
459  "system");
460 #else
461  if (simulation_precision_ == Precision::double_precision) {
462  // Double-precision density matrix simulation
463  return run_circuit_helper<
464  DensityMatrix::State<QV::DensityMatrixThrust<double>>>(
465  circ, noise, config, shots, rng_seed, cvector_t(),
466  Method::density_matrix_thrust_cpu);
467  } else {
468  // Single-precision density matrix simulation
469  return run_circuit_helper<
470  DensityMatrix::State<QV::DensityMatrixThrust<float>>>(
471  circ, noise, config, shots, rng_seed, cvector_t(),
472  Method::density_matrix_thrust_cpu);
473  }
474 #endif
475  case Method::stabilizer:
476  // Stabilizer simulation
477  // TODO: Stabilizer doesn't yet support custom state initialization
478  return run_circuit_helper<Stabilizer::State>(
479  circ, noise, config, shots, rng_seed, Clifford::Clifford(),
480  Method::stabilizer);
481  case Method::extended_stabilizer:
482  return run_circuit_helper<ExtendedStabilizer::State>(
483  circ, noise, config, shots, rng_seed, CHSimulator::Runner(),
484  Method::extended_stabilizer);
485 
486  case Method::matrix_product_state:
487  return run_circuit_helper<MatrixProductState::State>(
488  circ, noise, config, shots, rng_seed, MatrixProductState::MPS(),
489  Method::matrix_product_state);
490 
491  default:
492  throw std::runtime_error("QasmController:Invalid simulation method");
493  }
494 }

500 QasmController::Method QasmController::simulation_method(
501  const Circuit &circ, const Noise::NoiseModel &noise_model,
502  bool validate) const {
503  // Check simulation method and validate state
504  switch (simulation_method_) {
505  case Method::statevector: {
506  if (validate) {
507  if (simulation_precision_ == Precision::single_precision) {
508  Statevector::State<QV::QubitVector<float>> state;
509  validate_state(state, circ, noise_model, true);
510  } else {
511  Statevector::State<QV::QubitVector<>> state;
512  validate_state(state, circ, noise_model, true);
513  }
514  }
515  return Method::statevector;
516  }
517  case Method::statevector_thrust_gpu: {
518 #ifndef AER_THRUST_CUDA
519  throw std::runtime_error(
520  "QasmController: method statevector_gpu is not supported on this "
521  "system");
522 #else
523  if (validate) {
524  if (simulation_precision_ == Precision::single_precision) {
525  Statevector::State<QV::QubitVectorThrust<float>> state;
526  validate_state(state, circ, noise_model, true);
527  } else {
528  Statevector::State<QV::QubitVectorThrust<>> state;
529  validate_state(state, circ, noise_model, true);
530  }
531  }
532  return Method::statevector_thrust_gpu;
533 #endif
534  }
535  case Method::statevector_thrust_cpu: {
536 #ifndef AER_THRUST_CPU
537  throw std::runtime_error(
538  "QasmController: method statevector_thrust is not supported on this "
539  "system");
540 #else
541  if (validate) {
542  if (simulation_precision_ == Precision::single_precision) {
543  Statevector::State<QV::QubitVectorThrust<float>> state;
544  validate_state(state, circ, noise_model, true);
545  } else {
546  Statevector::State<QV::QubitVectorThrust<>> state;
547  validate_state(state, circ, noise_model, true);
548  }
549  }
550  return Method::statevector_thrust_cpu;
551 #endif
552  }
553  case Method::density_matrix: {
554  if (validate) {
555  if (simulation_precision_ == Precision::single_precision) {
556  validate_state(DensityMatrix::State<QV::DensityMatrix<float>>(), circ,
557  noise_model, true);
558  } else {
559  validate_state(DensityMatrix::State<QV::DensityMatrix<double>>(),
560  circ, noise_model, true);
561  }
562  }
563  return Method::density_matrix;
564  }
565  case Method::density_matrix_thrust_gpu: {
566 #ifndef AER_THRUST_SUPPORTED
567  throw std::runtime_error(
568  "QasmController: method density_matrix_gpu is not supported on this "
569  "system");
570 #else
571  if (validate) {
572  if (simulation_precision_ == Precision::single_precision) {
573  validate_state(DensityMatrix::State<QV::DensityMatrixThrust<float>>(),
574  circ, noise_model, true);
575  } else {
576  validate_state(
577  DensityMatrix::State<QV::DensityMatrixThrust<double>>(), circ,
578  noise_model, true);
579  }
580  }
581  return Method::density_matrix_thrust_gpu;
582 #endif
583  }
584  case Method::density_matrix_thrust_cpu: {
585 #ifndef AER_THRUST_SUPPORTED
586  throw std::runtime_error(
587  "QasmController: method density_matrix_thrust is not supported on this "
588  "system");
589 #else
590  if (validate) {
591  if (simulation_precision_ == Precision::single_precision) {
592  validate_state(DensityMatrix::State<QV::DensityMatrixThrust<float>>(),
593  circ, noise_model, true);
594  } else {
595  validate_state(
596  DensityMatrix::State<QV::DensityMatrixThrust<double>>(), circ,
597  noise_model, true);
598  }
599  }
600  return Method::density_matrix_thrust_cpu;
601 #endif
602  }
603  case Method::stabilizer: {
604  if (validate)
605  validate_state(Stabilizer::State(), circ, noise_model, true);
606  return Method::stabilizer;
607  }
608  case Method::extended_stabilizer: {
609  if (validate)
610  validate_state(ExtendedStabilizer::State(), circ, noise_model, true);
611  return Method::extended_stabilizer;
612  }
613  case Method::matrix_product_state: {
614  if (validate)
615  validate_state(MatrixProductState::State(), circ, noise_model, true);
616  return Method::matrix_product_state;
617  }
618  case Method::automatic: {
619  // If circuit and noise model are Clifford run on Stabilizer simulator
620  if (validate_state(Stabilizer::State(), circ, noise_model, false)) {
621  return Method::stabilizer;
622  }
623  // For noisy simulations we enable the density matrix method if
624  // shots > 2 ** num_qubits. This is based on a rough estimate that
625  // a single shot of the density matrix simulator is approx 2 ** nq
626  // times slow than a single shot of statevector due the increased
627  // dimension
628  if (noise_model.has_quantum_errors() &&
629  circ.shots > (1 << circ.num_qubits) &&
630  validate_memory_requirements(DensityMatrix::State<>(), circ, false) &&
631  validate_state(DensityMatrix::State<>(), circ, noise_model, false) &&
632  check_measure_sampling_opt(circ, Method::density_matrix)) {
633  return Method::density_matrix;
634  }
635  // Finally we check the statevector memory requirement for the
636  // current number of qubits. If it fits in available memory we
637  // default to the Statevector method. Otherwise we raise an exception
638  // and suggest using one of the other simulation methods.
639  bool enough_memory = true;
640  if (simulation_precision_ == Precision::single_precision) {
641  Statevector::State<QV::QubitVector<float>> sv_state;
642  enough_memory = validate_memory_requirements(sv_state, circ, false);
643  } else {
644  Statevector::State<> sv_state;
645  enough_memory = validate_memory_requirements(sv_state, circ, false);
646  }
647  if (!enough_memory) {
648  throw std::runtime_error(
649  "QasmSimulator: Insufficient memory for " + std::to_string(circ.num_qubits) + "-qubit"
650  R"( circuit using "statevector" method. You could try using the)"
651  R"( "matrix_product_state" or "extended_stabilizer" method instead.)"
652  );
653  }
654  }
655  // If we didn't select extended stabilizer above proceed to the default
656  // switch clause
657  default: {
658  // For default we use statevector followed by density matrix (for the case
659  // when the circuit contains invalid instructions for statevector)
660  if (validate_state(Statevector::State<>(), circ, noise_model, false)) {
661  return Method::statevector;
662  }
663  // If circuit contains invalid instructions for statevector throw a hail
664  // mary and try for density matrix.
665  if (validate)
666  validate_state(DensityMatrix::State<>(), circ, noise_model, true);
667  return Method::density_matrix;
668  }
669  }

790  const Circuit &circ, const Noise::NoiseModel &noise, const json_t &config,
791  uint_t shots, uint_t rng_seed, const Initstate_t &initial_state,
792  const Method method) const {
793  // Initialize new state object
794  State_t state;
795 
796  // Check memory requirements, raise exception if they're exceeded
797  validate_memory_requirements(state, circ, true);
798 
799  // Set state config
800  state.set_config(config);
801  state.set_parallalization(parallel_state_update_);
802 
803  // Rng engine
804  RngEngine rng;
805  rng.set_seed(rng_seed);
806 
807  // Output data container
808  ExperimentData data;
809  data.set_config(config);
810  data.add_metadata("method", state.name());
811  // Add measure sampling to metadata

813  data.add_metadata("measure_sampling", false);
814 
815  // Choose execution method based on noise and method
816  if (noise.is_ideal()) {
817  run_circuit_without_noise(circ, config, shots, state, initial_state, method, data, rng);
818  } else if ((method == Method::density_matrix ||
819  method == Method::density_matrix_thrust_gpu ||
820  method == Method::density_matrix_thrust_cpu) &&
821  noise.has_quantum_errors()) {

823  // and then execute the resulting circuit containing superoperators
824  Noise::NoiseModel noise_cpy = noise;
825  noise_cpy.activate_superop_method();
826  Circuit noise_circ = noise_cpy.sample_noise(circ, rng);
827  run_circuit_without_noise(noise_circ, config, shots, state, initial_state, method, data, rng);
828  } else if (noise.has_quantum_errors() == false) {
829  // We can insert the readout errors from the noise model and then
830  // execute the resulting circuit
831  Circuit noise_circ = noise.sample_noise(circ, rng);
832  run_circuit_without_noise(noise_circ, config, shots, state, initial_state, method, data, rng);
833  } else {
834  // Run sampling a noisy instance of the circuit for each shot
835  run_circuit_with_noise(circ, noise, config, shots, state, initial_state, method, data, rng);
836  }
837  return data;

873 template <class State_t, class Initstate_t>
874 void QasmController::run_circuit_without_noise(const Circuit &circ,
875  const json_t &config, uint_t shots, State_t &state,
876  const Initstate_t &initial_state,
877  const Method method,
878  ExperimentData &data,
879  RngEngine &rng) const {
880  // Optimize circuit for state type
881  Circuit opt_circ = circ;
882 
883  // Dummy noise model for transpiler passes
884  Noise::NoiseModel dummy_noise;
885 
886  // Apply fusion transpilation pass
887  auto fusion_pass = transpile_fusion(method, config);
888  fusion_pass.optimize_circuit(opt_circ, dummy_noise, state.opset(), data);
889 
890  // Apply delay measure transpilation pass
891  Transpile::DelayMeasure measure_pass;
892  measure_pass.set_config(config);
893  measure_pass.optimize_circuit(opt_circ, dummy_noise, state.opset(), data);
894

904  state.apply_ops(ops, data, rng);

907  ops = std::vector<Operations::Op>(opt_circ.ops.begin() + pos,
908  opt_circ.ops.end());
909  measure_sampler(ops, shots, state, data, rng);
910

912  data.add_metadata("measure_sampling", true);
913  } else {
914  // Perform standard execution if we cannot apply the
915  // measurement sampling optimization
916  while (shots-- > 0) {
917  run_single_shot(opt_circ, state, initial_state, data, rng);
918  }
919  }

926 bool QasmController::check_measure_sampling_opt(
927  const Circuit &circ, const Method method) const {
928  // Check if circuit has sampling flag disabled
929  if (circ.can_sample == false) {
930  return false;
931  }
932 
933  // Check if stabilizer measure sampling has been disabled
934  if (method == Method::extended_stabilizer &&
935  !extended_stabilizer_measure_sampling_) {
936  return false;
937  }
938 
939  // Check if non-density matrix simulation and circuit contains
940  // a stochastic instruction before measurement
941  // ie. initialize, reset, kraus, superop, conditional
942  // TODO:
943  // * If initialize should be allowed if applied to product states (ie start of circuit)
944  // * Resets should be allowed if applied to |0> state (no gates before).
945  bool density_mat = (method == Method::density_matrix ||
946  method == Method::density_matrix_thrust_gpu ||
947  method == Method::density_matrix_thrust_cpu);
948  if (!density_mat && (
949  circ.opset().contains(Operations::OpType::reset) ||
950  circ.opset().contains(Operations::OpType::initialize) ||
951  circ.opset().contains(Operations::OpType::kraus) ||
952  circ.opset().contains(Operations::OpType::superop))) {
953  return false;
954  }

956  return true;

959 template <class State_t>
960 void QasmController::measure_sampler(
961  const std::vector<Operations::Op> &meas_roerror_ops, uint_t shots,
962  State_t &state, ExperimentData &data, RngEngine &rng) const {
963  // Check if meas_circ is empty, and if so return initial creg
964  if (meas_roerror_ops.empty()) {
965  while (shots-- > 0) {
966  state.add_creg_to_data(data);
967  }
968  return;
969  }
970 
971  std::vector<Operations::Op> meas_ops;
972  std::vector<Operations::Op> roerror_ops;
973  for (const Operations::Op &op : meas_roerror_ops)
974  if (op.type == Operations::OpType::roerror)
975  roerror_ops.push_back(op);
976  else /*(op.type == Operations::OpType::measure) */
977  meas_ops.push_back(op);
978 
979  // Get measured qubits from circuit sort and delete duplicates
980  std::vector<uint_t> meas_qubits; // measured qubits
981  for (const auto &op : meas_ops) {
982  for (size_t j = 0; j < op.qubits.size(); ++j)
983  meas_qubits.push_back(op.qubits[j]);
984  }
985  sort(meas_qubits.begin(), meas_qubits.end());
986  meas_qubits.erase(unique(meas_qubits.begin(), meas_qubits.end()),
987  meas_qubits.end());
988  // Generate the samples
989  auto all_samples = state.sample_measure(meas_qubits, shots, rng);
990 
991  // Make qubit map of position in vector of measured qubits
992  std::unordered_map<uint_t, uint_t> qubit_map;
993  for (uint_t j = 0; j < meas_qubits.size(); ++j) {
994  qubit_map[meas_qubits[j]] = j;
995  }
996 
997  // Maps of memory and register to qubit position
998  std::unordered_map<uint_t, uint_t> memory_map;
999  std::unordered_map<uint_t, uint_t> register_map;
1000  for (const auto &op : meas_ops) {
1001  for (size_t j = 0; j < op.qubits.size(); ++j) {
1002  auto pos = qubit_map[op.qubits[j]];
1003  if (!op.memory.empty()) memory_map[op.memory[j]] = pos;
1004  if (!op.registers.empty()) register_map[op.registers[j]] = pos;
1005  }
1006  }
1007 
1008  // Process samples
1009  // Convert opts to circuit so we can get the needed creg sizes
1010  // NB: this function could probably be moved somewhere else like Utils or Ops
1011  Circuit meas_circ(meas_roerror_ops);
1012  ClassicalRegister creg;
1013  while (!all_samples.empty()) {
1014  auto sample = all_samples.back();
1015  creg.initialize(meas_circ.num_memory, meas_circ.num_registers);
1016 
1017  // process memory bit measurements
1018  for (const auto &pair : memory_map) {
1019  creg.store_measure(reg_t({sample[pair.second]}), reg_t({pair.first}),
1020  reg_t());
1021  }
1022  // process register bit measurements
1023  for (const auto &pair : register_map) {
1024  creg.store_measure(reg_t({sample[pair.second]}), reg_t(),
1025  reg_t({pair.first}));
1026  }
1027 
1028  // process read out errors for memory and registers
1029  for (const Operations::Op &roerror : roerror_ops) {
1030  creg.apply_roerror(roerror, rng);
1031  }
1032 
1033  auto memory = creg.memory_hex();
1034  data.add_memory_count(memory);
1035  data.add_pershot_memory(memory);
1036 
1037  data.add_pershot_register(creg.register_hex());
1038 
1039  // pop off processed sample
1040  all_samples.pop_back();
1041  }
1042 }
