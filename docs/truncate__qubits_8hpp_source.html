80 void TruncateQubits::optimize_circuit(Circuit& circ,
81  Noise::NoiseModel& noise,
82  const Operations::OpSet &allowed_opset,
83  ExperimentData &data) const {
84 
85  // Check if circuit operations allow remapping
86  // Remapped circuits must return the same output data as the
87  // original circuit
88  if (!active_ || !can_apply(circ))
89  return;
90 
91  // Get qubits actually used in the circuit
92  // If this is all qubits we don't need to remap
93  reg_t active_qubits = get_active_qubits(circ, noise);
94  if (active_qubits.size() == circ.num_qubits)
95  return;
96 
97  // Generate the qubit mapping {original_qubit: new_qubit}
98  mapping_t mapping = generate_mapping(active_qubits, circ, noise);
99 
100  // Remap circuit operations
101  for (Operations::Op& op: circ.ops) {
102  remap_qubits(op.qubits, mapping);
103  // Remap regs
104  for (reg_t &reg : op.regs) {
105  remap_qubits(reg, mapping);
106  }
107  }
108  // Update the number of qubits in the circuit
109  circ.num_qubits = active_qubits.size();
110 
111  // Remap noise model
112  noise.remap_qubits(mapping);
113 
114  if (verbose_) {
115  json_t truncate_metadata;
116  truncate_metadata["active_qubits"] = active_qubits;
117  truncate_metadata["mapping"] = mapping;
118  data.add_metadata("truncate_qubits", truncate_metadata);
119  }
120 }

122 reg_t TruncateQubits::get_active_qubits(const Circuit& circ,
123  const Noise::NoiseModel& noise) const {
124 
125  size_t not_used = circ.num_qubits + 1;
126  reg_t active_qubits = reg_t(circ.num_qubits, not_used);
127 
128  for (const Operations::Op& op: circ.ops) {
129  for (size_t qubit: op.qubits)
130  active_qubits[qubit] = qubit;
131  for (const reg_t &reg: op.regs)
132  for (size_t qubit: reg)
133  active_qubits[qubit] = qubit;
134 
135  // Add noise model qubits
136  // The label for checking noise is either stored in string_params
137  // or is the op name
138  std::string label = "";
139  if (op.string_params.size() == 1) {
140  label = op.string_params[0];
141  }
142  if (label == "")
143  label = op.name;
144  const auto noise_reg = noise.nonlocal_noise_qubits(label, op.qubits);
145  for (size_t qubit: noise_reg) {
146  // A noise model might have more qubits in it than are in
147  // the original circuit. In this case we only add qubits
148  // up to the number of qubits in the circuit
149  if (qubit < circ.num_qubits)
150  active_qubits[qubit] = qubit;
151  }
152  }
153 
154  // Erase unused qubits for the list
155  active_qubits.erase(std::remove(active_qubits.begin(), active_qubits.end(), not_used),
156  active_qubits.end());
157  return active_qubits;
158 }

160 TruncateQubits::mapping_t
161 TruncateQubits::generate_mapping(const reg_t& active_qubits,
162  const Circuit& circ,
163  const Noise::NoiseModel& noise) const {
164  // Convert to mapping
165  mapping_t mapping;
166  for (const auto & qubit : active_qubits) {
167  size_t new_qubit = std::distance(active_qubits.begin(),
168  find(active_qubits.begin(),
169  active_qubits.end(), qubit));
170  mapping[qubit] = new_qubit;
171  }
172 
173  // Now we need to complete the mapping by adding qubits not in the input space
174  // This is required for remapping the noise model.
175  if (!noise.is_ideal()) {
176  uint_t unused_qubit = active_qubits.size();
177  for (uint_t qubit=0; qubit<circ.num_qubits; qubit++) {
178  if (mapping.find(qubit) == mapping.end()) {
179  mapping[qubit] = unused_qubit;
180  unused_qubit++; // increment unused qubit position
181  }
182  }
183  }
184  return mapping;
185 }

201 bool TruncateQubits::can_apply(const Operations::Op& op) const {
