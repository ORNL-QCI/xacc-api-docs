323 void StabilizerState::X(unsigned q)
324 {
325  //Commute the Pauli through UC
326  if (!isReadyMT)
327  {
328  TransposeM();
329  }
330  if (!isReadyFT)
331  {
332  TransposeF();
333  }
334  uint_fast64_t x_string = FT[q];
335  uint_fast64_t z_string = MT[q];
336  //Initial phase correction
337  int phase = 2*((gamma1 >> q)&one) + 4*((gamma2>>q) & one);
338  //Commute the z_string through the hadamard layer
339  // Each z that hits a hadamard becomes a Pauli X
340  s ^= (z_string & v);
341  //Remaining Z gates add a global phase from their action on the s string
342  phase += 4*(hamming_parity((z_string & ~(v)) & s));
343  //Commute the x_string through the hadamard layer
344  // Any remaining X gates update s
345  s ^= (x_string & ~(v));
346  //New z gates add a global phase from their action on the s string
347  phase += 4*(hamming_parity((x_string & v) & s));
348  //Update the global phase
349  omega.e = (omega.e + phase)%8;
350 }

367 void StabilizerState::CX(unsigned q, unsigned r)
368 {
369  isReadyMT=false;// we are going to change M and F
370  isReadyFT=false;
371  uint_fast64_t C=(one<<q);
372  uint_fast64_t T=(one<<r);
373  bool b=false;
374  for (unsigned p=0; p<n; p++)
375  {
376  b=( b != ( ( M[p] & C ) && ( F[p] & T ) ) );
377  G[p]^=((G[p]>>q) & one)*T;
378  F[p]^=((F[p]>>r) & one)*C;
379  M[p]^=((M[p]>>r) & one)*C;
380  }
381  // update phase vector as
382  // gamma[q] gets gamma[q] + gamma[r] + 2*b (mod 4)
383  if (b) gamma2^=C;
384  b= ( (gamma1 & C) && (gamma1 & T));
385  gamma1^=((gamma1 >> r) & one)*C;
386  gamma2^=((gamma2 >> r) & one)*C;
387  if (b) gamma2^=C;

411 {

413  if (!isReadyMT)

438 scalar_t StabilizerState::Amplitude(uint_fast64_t x)
439 {
440  // compute transposed matrices if needed
441  if (!isReadyMT)
442  {
443  TransposeM();
444  }
445  if (!isReadyFT)
446  {
447  TransposeF();
448  }
449  // compute Pauli U_C^{-1} X(x) U_C
450  P=GetPauliX(x);
451 
452  if (!omega.eps) return omega; // the state is zero
453 
454  // now the amplitude = complex conjugate of <s|U_H P |0^n>
455  // Z-part of P is absorbed into 0^n
456 
457  scalar_t amp;
458  amp.e=2*P.e;
459  int p = (int) hamming_weight(v);
460  amp.p= -1 * p;// each Hadamard gate contributes 1/sqrt(2)
461  bool isNonZero=true;
462 
463 
464  for (unsigned q=0; q<n; q++)
465  {
466  uint_fast64_t pos=(one<<q);
467  if (v & pos)
468  {
469  amp.e += 4*( (s & pos) && (P.X & pos) ); // minus sign that comes from <1|H|1>
470  }
471  else
472  {
473  isNonZero=( ((P.X & pos)==(s & pos)) );
474  }
475  if (!isNonZero) break;
476  }
477 
478 
479  amp.e%=8;
480  if (isNonZero)
481  {
482  amp.conjugate();
483  }
484  else
485  {
486  amp.eps=0;
487  return amp;
488  }
489 
490  // multiply amp by omega
491  amp.p+=omega.p;
492  amp.e=(amp.e + omega.e) % 8;
493  return amp;
494 }

497 {
498  // Q gets Pauli operator U_C^{-1} X_{flip_pos} U_C
499  Q.e=1*((gamma1>>flip_pos) & one);
500  Q.e+=2*((gamma2>>flip_pos) & one);
501  Q.X=FT[flip_pos];
502  Q.Z=MT[flip_pos];
503  Q*=P;
504 
505  if (!omega.eps) return omega; // the state is zero
506 
507  // the rest is the same as Amplitude() except that P becomes Q
508 
509  // now the amplitude = complex conjugate of <s|U_H Q |0^n>

514  scalar_t amp;

516  amp.p=-1*(hamming_weight(v));// each Hadamard gate contributes 1/sqrt(2)

522  if (v & pos)
523  {

526  else

531  }

604 void StabilizerState::UpdateSvector(uint_fast64_t t, uint_fast64_t u, unsigned b)
605 {
606 // take care of the trivial case: t=u
607  if (t==u) // multiply omega by (1+i^b)/sqrt(2)
608  switch(b)
609  {
610  case 0 :
611  omega.p+=1;
612  s=t;
613  return;
614  case 1 :
615  s=t;
616  omega.e=(omega.e + 1) % 8;
617  return;
618  case 2 :
619  s=t;
620  omega.eps=0;
621  return;
622  case 3 :
623  s=t;
624  omega.e=(omega.e + 7) % 8;
625  return;
626  default :
627  // we should not get here
628  throw std::logic_error("Invalid phase factor found b:" + std::to_string(b) + ".\n");
629  }
630 
631  // now t and u are distinct
632  isReadyFT=false; // below we are going to change F and M
633  isReadyMT=false;
634  // naming of variables roughly follows Section IIIA
635  uint_fast64_t ut=u^t;
636  uint_fast64_t nu0 = (~v) & ut;
637  uint_fast64_t nu1 = v & ut;
638  //
639  b%=4;
640  unsigned q=0;
641  uint_fast64_t qpos=zer;
642  if (nu0)
643  {
644 
645  // the subset nu0 is non-empty
646  // find the first element of nu0
647  q=0;
648  while (!(nu0 & (one<<q))) q++;
649 
650  qpos=(one<<q);
651  if(!(nu0 & qpos))
652  {
653  throw std::logic_error("Failed to find first bit of nu despite being non-empty.");
654  }
655 
656  // if nu0 has size >1 then multiply U_C on the right by the first half of the circuit VC
657  nu0^=qpos; // set q-th bit to zero
658  if (nu0)
659  for (unsigned q1=q+1; q1<n; q1++)
660  if (nu0 & (one<<q1))
661  RightCX(q,q1);
662 
663  // if nu1 has size >0 then apply the second half of the circuit VC
664  if (nu1)
665  for (unsigned q1=0; q1<n; q1++)
666  if (nu1 & (one<<q1))
667  RightCZ(q,q1);
668 
669  }// if (nu0)
670  else
671  {
672  // if we got here when nu0 is empty
673  // find the first element of nu1
674  q=0;
675  while (!(nu1 & (one<<q))) q++;
676 
677  qpos=(one<<q);
678  if(!(nu1 & qpos))
679  {
680  throw std::logic_error("Expect at least one non-zero element in nu1.\n");
681  }
682 
683  // if nu1 has size >1 then apply the circuit VC
684  nu1^=qpos;
685  if (nu1)
686  for (unsigned q1=q+1; q1<n; q1++)
687  if (nu1 & (one<<q1))
688  RightCX(q1,q);
689 
690  }// if (nu0) else
691 
692  // update the initial state
693  // if t_q=1 then switch t_q and u_q
694  // uint_fast64_t y,z;
695  if (t & qpos)
696  {
697  s=u;
698  omega.e=(omega.e + 2*b) % 8;
699  b=(4-b) % 4;
700  if(!!(u & qpos))
701  {
702 
703  }
704  }
705  else
706  s=t;
707 
708  // change the order of H and S gates
709  // H^{a} S^{b} |+> = eta^{e1} S^{e2} H^{e3} |e4>
710  // here eta=exp( i (pi/4) )
711  // a=0,1
712  // b=0,1,2,3
713  //
714  // H^0 S^0 |+> = eta^0 S^0 H^1 |0>
715  // H^0 S^1 |+> = eta^0 S^1 H^1 |0>
716  // H^0 S^2 |+> = eta^0 S^0 H^1 |1>
717  // H^0 S^3 |+> = eta^0 S^1 H^1 |1>
718  //
719  // H^1 S^0 |+> = eta^0 S^0 H^0 |0>
720  // H^1 S^1 |+> = eta^1 S^1 H^1 |1>
721  // H^1 S^2 |+> = eta^0 S^0 H^0 |1>
722  // H^1 S^3 |+> = eta^{7} S^1 H^1 |0>
723  //
724  // "analytic" formula:
725  // e1 = a * (b mod 2) * ( 3*b -2 )
726  // e2 = b mod 2
727  // e3 = not(a) + a * (b mod 2)
728  // e4 = not(a)*(b>=2) + a*( (b==1) || (b==2) )
729 
730  bool a=((v & qpos)>0);
731  unsigned e1=a*(b % 2)*( 3*b -2);
732  unsigned e2 = b % 2;
733  bool e3 = ( (!a) != (a && ((b % 2)>0) ) );
734  bool e4 = ( ( (!a) && (b>=2) ) != (a && ((b==1) || (b==2))) );
735 
736  // update CH-form
737  // set q-th bit of s to e4
738  s&=~qpos;
739  s^=e4*qpos;
740 
741  // set q-th bit of v to e3
742  v&=~qpos;
743  v^=e3*qpos;
744 
745  // update the scalar factor omega
746  omega.e=(omega.e + e1) % 8;
747 
748  // multiply the C-layer on the right by S^{e2} on the q-th qubit
749  if (e2) RightS(q);

754 void StabilizerState::H(unsigned q)
755 {
756  isReadyMT=false;// we are going to change M and F
757  isReadyFT=false;
758  // extract the q-th row of F,G,M
759  uint_fast64_t rowF=zer;
760  uint_fast64_t rowG=zer;
761  uint_fast64_t rowM=zer;
762  for (unsigned j=0; j<n; j++)
763  {
764  uint_fast64_t pos=(one<<j);
765  rowF^=pos*( (F[j]>>q) & one );
766  rowG^=pos*( (G[j]>>q) & one );
767  rowM^=pos*( (M[j]>>q) & one );
768  }
769 
770  // after commuting H through the C and H laters it maps |s> to a state
771  // sqrt(0.5)*[ (-1)^alpha |t> + i^{gamma[p]} (-1)^beta |u> ]
772  //
773  // compute t,s,alpha,beta
774  uint_fast64_t t = s ^ (rowG & v);
775  uint_fast64_t u = s ^ (rowF & (~v)) ^ (rowM & v);
776 
777  unsigned alpha = hamming_weight( rowG & (~v) & s );
778  unsigned beta = hamming_weight( (rowM & (~v) & s) ^ (rowF & v & (rowM ^ s)) );
779 
780  if (alpha % 2) omega.e=(omega.e+4) % 8;
781  // get the phase gamma[q]
782  unsigned phase = ((gamma1>>q) & one) + 2*((gamma2>>q) & one);
783  unsigned b=(phase + 2*alpha + 2*beta) % 4;
784 
785 
786  // now the initial state is sqrt(0.5)*(|t> + i^b |u>)
787 
788  // take care of the trivial case
789  if (t==u)
790  {
791  s=t;
792  if(!((b==1) || (b==3))) // otherwise the state is not normalized
793  {
794  throw std::logic_error("State is not properly normalised, b should be 1 or 3.\n");
795  }
796  if (b==1)
797  omega.e=(omega.e + 1) % 8;
798  else
799  omega.e=(omega.e + 7) % 8;
800  }
801  else
802  UpdateSvector(t,u,b);
803

806 void StabilizerState::MeasurePauli(pauli_t PP)
807 {
808  // compute Pauli R = U_C^{-1} P U_C
809  pauli_t R;
810  R.e = PP.e;
811 
812  for (unsigned j=0; j<n; j++)
813  if ((PP.X>>j) & one)
814  {
815  // multiply R by U_C^{-1} X_j U_C
816  // extract the j-th rows of F and M
817  uint_fast64_t rowF=zer;
818  uint_fast64_t rowM=zer;
819  for (unsigned i=0; i<n; i++)
820  {
821  rowF^=(one<<i)*( (F[i]>>j) & one);
822  rowM^=(one<<i)*( (M[i]>>j) & one);
823  }
824  R.e+= 2*hamming_weight(R.Z & rowF); // extra sign from Pauli commutation
825  R.Z^=rowM;
826  R.X^=rowF;
827  R.e+= ((gamma1>>j) & one) + 2*((gamma2>>j) & one);
828  }
829  for (unsigned q=0; q<n; q++)
830  R.Z^=(one<<q)*(hamming_weight( PP.Z & G[q]) % 2);
831 
832  // now R=U_C^{-1} PP U_C
833  // next conjugate R by U_H
834  uint_fast64_t tempX = ( (~v) & R.X ) ^ (v & R.Z);
835  uint_fast64_t tempZ = ( (~v) & R.Z ) ^ (v & R.X);
836  // the sign flips each time a Hadamard hits Y on some qubit
837  R.e=(R.e + 2*hamming_weight(v & R.X & R.Z)) % 4;
838  R.X=tempX;
839  R.Z=tempZ;
840 
841  // now the initial state |s> becomes 0.5*(|s> + R |s>) = 0.5*(|s> + i^b |s ^ R.X>)
842  unsigned b = (R.e + 2*hamming_weight(R.Z & s) ) % 4;
843  UpdateSvector(s, s ^ R.X, b);
844  // account for the extra factor sqrt(1/2)
845  omega.p-=1;
846 
847  isReadyMT=false;// we have changed change M and F
848  isReadyFT=false;
849 }

864 scalar_t StabilizerState::InnerProduct(const uint_fast64_t& A_diag1,
865  const uint_fast64_t& A_diag2,
866  const std::vector<uint_fast64_t>& A)
867 {
868  uint_fast64_t K_diag1 = zer, K_diag2=zer, J_diag1=gamma1, J_diag2=gamma2;
869  std::vector<uint_fast64_t> J(n, zer);
870  std::vector<uint_fast64_t> K(n, zer);
871  std::vector<uint_fast64_t> placeholder(n, zer);
872  if(!isReadyMT)
873  {
874  TransposeM();
875  }
876  if (!isReadyFT)
877  {
878  TransposeF();
879  }
880  //Setup the J matrix
881  for (size_t i=0; i<n; i++)
882  {
883  for (size_t j=i; j<n; j++)
884  {
885  if (hamming_parity(MT[i] & FT[j]))
886  {
887  J[i] |= (one << j);
888  J[j] |= (one << i);
889  }
890  }
891  }
892  //Calculate the matrix J = A+J
893  J_diag2 = (A_diag2 ^ J_diag2 ^ (A_diag1&J_diag1));
894  J_diag1 = (A_diag1 ^ J_diag1);
895  for (size_t i=0; i<n; i++)
896  {
897  J[i] ^= A[i];
898  }
899  //placeholder = J*G)
900  for (size_t i=0; i<n; i++)
901  {
902  //Grab column i of J, it's symmetric
903  uint_fast64_t col_i = J[i];
904  for (size_t j=0; j<n; j++)
905  {
906  if (hamming_parity(col_i & G[j]))
907  {
908  placeholder[j] |= (one<<i);
909  }
910  }
911  }
912  //K = GT*placeholder
913  for (size_t i=0; i<n; i++)
914  {
915  uint_fast64_t col_i = G[i];
916  uint_fast64_t shift = (one << i);
917  for (size_t j=i; j<n; j++)
918  {
919  if(hamming_parity(col_i & placeholder[j]))
920  {
921  K[j] |= shift;
922  K[i] |= (one << j);
923  }
924  }
925  for (size_t r =0; r<n; r++)
926  {
927  if ((col_i >> r) & one)
928  {
929  uint_fast64_t one_bit = (J_diag1 >> r) & one;
930  if ((K_diag1 >> i) & one_bit)
931  {
932  K_diag2 ^= shift;
933  }
934  K_diag1 ^= one_bit*shift;
935  K_diag2 ^= ((J_diag2 >> r)&one) * shift;
936  }
937  for (size_t k=r+1; k<n; k++)
938  {
939  if ((J[k] >> r) & (col_i >> r) & (col_i >> k) & one)
940  {
941  K_diag2 ^= shift;
942  }
943  }
944  }
945  }
946  unsigned col=0;
947  QuadraticForm q(hamming_weight(v));
948  //We need to setup a quadratic form to evaluate the Exponential Sum
949  for (size_t i=0; i<n; i++)
950  {
951  if ((v>>i) & one)
952  {
953  uint_fast64_t shift = (one << col);
954  //D = Diag(K(1,1)) + 2*[s + s*K](1)
955  // J = K(1,1);
956  q.D1 ^= ((K_diag1 >> i) & one) * shift;
957  q.D2 ^= (( ((K_diag2 >> i) ^ (s >> i)) & one)
958  ^ hamming_parity(K[i] & s)) * shift;
959  // q.D2 ^= ((s >> i) & one) * shift;
960  // q.D2 ^= hamming_parity(K[i] & s) * shift;
961  unsigned row=0;
962  for (size_t j=0; j<n; j++)
963  {
964  if((v>>j) & one)
965  {
966  q.J[col] |= (((K[i] >> j) & one) << row);
967  row++;
968  }
969  }
970  col++;
971  }
972  }
973  // Q = 4* (s.v) + sKs
974  q.Q = hamming_parity(s&v)*4;
975  for (size_t i=0; i<n; i++)
976  {
977  if ((s>>i) & one)
978  {
979  q.Q = (q.Q + 4*((K_diag2 >> i) & one) + 2*((K_diag1 >> i) & one))%8;// + 2*((K_diag1 >> i) & one))%8;
980  for (size_t j=i+1; j<n; j++)
981  {
982  if ((s>>j) & (K[j] >> i) & one)
983  {
984  q.Q ^= 4;
985  }
986  }
987  }
988  }
989  scalar_t amp = q.ExponentialSum();
990  // Reweight by 2^{-(n+|v|)}/2
991  amp.p -= (n+hamming_weight(v));
992  // We need to further multiply by omega*
993  scalar_t psi_amp(omega);
994  psi_amp.conjugate();
995  amp *= psi_amp;
996  return amp;
997 }
