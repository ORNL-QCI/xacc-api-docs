291 void Runner::apply_u1(uint_t qubit, complex_t param, double r, int rank)
292 {
293  double lambda = std::real(param);
294  auto it = Z_ROTATIONS.find(lambda); //Look for cached z_rotations
295  sample_branch_t branch;
296  if (it == Z_ROTATIONS.end())
297  {
298  U1Sample rotation(lambda);
299  Z_ROTATIONS.insert({lambda, rotation});
300  branch = rotation.sample(r);
301  }
302  else
303  {
304  branch = it->second.sample(r);
305  }
306  coefficients_[rank] *= branch.first;
307  switch(branch.second)
308  {
309  case Gates::s:
310  states_[rank].S(qubit);
311  break;
312  case Gates::sdg:
313  states_[rank].Sdag(qubit);
314  break;
315  case Gates::z:
316  states_[rank].Z(qubit);
317  break;
318  default:
319  break;
320  }
321 }

411 double Runner::norm_estimation(uint_t n_samples, AER::RngEngine &rng)
412 {
413  std::vector<uint_t> adiag_1(n_samples, 0ULL);
414  std::vector<uint_t> adiag_2(n_samples, 0ULL);
415  std::vector< std::vector<uint_t> > a(n_samples, std::vector<uint_t>(n_qubits_, 0ULL));
416  const int_t NSAMPLES = n_samples;
417  const int_t NQUBITS = n_qubits_;
418  #pragma omp parallel if (num_threads_ > 1) num_threads(num_threads_)
419  {
420  #ifdef _WIN32
421  #pragma omp for
422  #else
423  #pragma omp for collapse(2)
424  #endif
425  for (int_t l=0; l<NSAMPLES; l++)
426  {
427  for (int_t i=0; i<NQUBITS; i++)
428  {
429  for (int_t j=i; j<NQUBITS; j++)
430  {
431  if(rng.rand() < 0.5)
432  {
433  a[l][i] |= (1ULL << j);
434  a[l][j] |= (1ULL << i);
435  }
436  }
437  adiag_1[l] |= (a[l][i] & (1ULL << i));
438  if (rng.rand() < 0.5)
439  {
440  adiag_2[l] |= (1ULL << i);
441  }
442  }
443  }
444  } // end omp parallel
445  return ParallelNormEstimate(states_, coefficients_, adiag_1, adiag_2, a, num_threads_);
446 }

476 void Runner::init_metropolis(AER::RngEngine &rng)
477 {
478  accept_ = 0;
479  //Random initial x_string from RngEngine
480  uint_t max = (1ULL<<n_qubits_) - 1;
481  x_string_ = rng.rand_int(ZERO, max);
482  last_proposal_=0;
483  double local_real=0., local_imag=0.;
484  const int_t END = num_states_;
485  #pragma omp parallel for if(num_states_ > omp_threshold_ && num_threads_ > 1) num_threads(num_threads_) reduction(+:local_real) reduction(+:local_imag)
486  for (int_t i=0; i<END; i++)
487  {
488  scalar_t amp = states_[i].Amplitude(x_string_);
489  if(amp.eps == 1)
490  {
491  complex_t local = (amp.to_complex() * coefficients_[i]);
492  local_real += local.real();
493  local_imag += local.imag();
494  }
495  }
496  old_ampsum_ = complex_t(local_real, local_imag);
497 }

499 void Runner::metropolis_step(AER::RngEngine &rng)
500 {
501  uint_t proposal = rng.rand(0ULL, n_qubits_);
502  if(accept_)
503  {
504  x_string_ ^= (one << last_proposal_);
505  }
506  double real_part = 0.,imag_part =0.;
507  if (accept_ == 0)
508  {
509  const int_t END = num_states_;
510  #pragma omp parallel for if(num_states_ > omp_threshold_ && num_threads_ > 1) num_threads(num_threads_) reduction(+:real_part) reduction(+:imag_part)
511  for (int_t i=0; i<END; i++)
512  {
513  scalar_t amp = states_[i].ProposeFlip(proposal);
514  if(amp.eps == 1)
515  {
516  complex_t local = (amp.to_complex() * coefficients_[i]);
517  real_part += local.real();
518  imag_part += local.imag();
519  }
520  }
521  }
522  else
523  {
524  const int_t END = num_states_;
525  #pragma omp parallel for if(num_states_ > omp_threshold_ && num_threads_ > 1) num_threads(num_threads_) reduction(+:real_part) reduction(+:imag_part)
526  for (int_t i=0; i<END; i++)
527  {
528  states_[i].AcceptFlip();
529  scalar_t amp = states_[i].ProposeFlip(proposal);
530  if(amp.eps == 1)
531  {
532  complex_t local = (amp.to_complex() * coefficients_[i]);
533  real_part += local.real();
534  imag_part += local.imag();
535  }
536  }
537  }
538  complex_t ampsum(real_part, imag_part);
539  double p_threshold = std::norm(ampsum)/std::norm(old_ampsum_);
540  #ifdef __FAST_MATH__ //isnan doesn't behave well under fastmath, so use absolute tolerance check instead
541  if(std::isinf(p_threshold) || std::abs(std::norm(old_ampsum_)-0.) < 1e-8)
542  #else
543  if(std::isinf(p_threshold) || std::isnan(p_threshold))
544  #endif
545  {
546  accept_ = 1;
547  old_ampsum_ = ampsum;
548  last_proposal_ = proposal; //We try to move away from node with 0 probability.
549  }
550  else
551  {
552  double rand = rng.rand();
553  if (rand < p_threshold)
554  {
555  accept_ = 1;
556  old_ampsum_ = ampsum;
557  last_proposal_ = proposal;
558  }
559  else
560  {
561  accept_ = 0;
562  }
563  }
564 }

587 complex_t Runner::amplitude(uint_t x_measure)
588 {
589  double real_part=0., imag_part=0.;
590  //Splitting the reduction guarantees support on more OMP versions.
591  const int_t END = num_states_;
592  #pragma omp parallel for if(num_states_ > omp_threshold_ && num_threads_ > 1) num_threads(num_threads_) reduction(+:real_part) reduction(+:imag_part)
593  for(int_t i=0; i<END; i++)
594  {
595  complex_t amplitude = states_[i].Amplitude(x_measure).to_complex();
596  amplitude *= coefficients_[i];
597  real_part += amplitude.real();
598  imag_part += amplitude.imag();
599  }
600  return complex_t(real_part, imag_part);
601 }

646 json_t Runner::serialize_state(uint_t rank) const
647 {
648  json_t js = json_t::object();
649  std::vector<unsigned> gamma;
650  std::vector<uint_t> M;
651  std::vector<uint_t> F;
652  std::vector<uint_t> G;
653  gamma.reserve(n_qubits_);
654  M = states_[rank].MMatrix();
655  F = states_[rank].FMatrix();
656  G = states_[rank].GMatrix();
657  uint_t gamma1 = states_[rank].Gamma1();
658  uint_t gamma2 = states_[rank].Gamma2();
659  for(uint_t i=0; i<n_qubits_; i++)
660  {
661  gamma.push_back(((gamma1 >> i) & 1ULL) + 2*((gamma2 >> i) & 1ULL));
662  }
663  js["gamma"] = gamma;
664  js["M"] = M;
665  js["F"] = F;
666  js["G"] = G;
667  js["internal_cofficient"] = states_[rank].Omega().to_complex();
668  js["coefficient"] = coefficients_[rank];
669  return js;
670 }
