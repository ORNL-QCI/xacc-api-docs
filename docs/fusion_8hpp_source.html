147 void Fusion::optimize_circuit(Circuit& circ,
148  Noise::NoiseModel& noise,
149  const opset_t &allowed_opset,
150  ExperimentData &data) const {
151  // Check if fusion should be skipped
152  if (!active || !allowed_opset.contains(optype_t::matrix))
153  return;
154 
155  // Start timer
156  using clock_t = std::chrono::high_resolution_clock;
157  auto timer_start = clock_t::now();
158 
159  // Determine fusion method
160  // TODO: Support Kraus fusion method
161  Method method = Method::unitary;
162  if (allow_noise && allowed_opset.contains(optype_t::superop) &&
163  (circ.opset().contains(optype_t::kraus)
164  || circ.opset().contains(optype_t::superop)
165  || circ.opset().contains(optype_t::reset))) {
166  method = Method::superop;
167  }
168  uint_t qubit_threshold = (method==Method::unitary) ? threshold : threshold / 2;
169  uint_t max_fused_qubits = (method==Method::unitary) ? max_qubit : max_qubit / 2;
170 
171  // Fusion metadata container
172  json_t metadata;
173  metadata["applied"] = false;
174  if (method == Method::unitary) {
175  metadata["method"] = "unitary";
176  } else if (method == Method::superop) {
177  metadata["method"] = "superop";
178  } else if (method == Method::kraus) {
179  metadata["method"] = "kraus";
180  }
181  metadata["threshold"] = qubit_threshold;
182  metadata["cost_factor"] = cost_factor;
183  metadata["max_fused_qubits"] = max_fused_qubits;
184 
185  // If we are doing superoperator fusion we have the threshold
186  // for the density matrix simulator
187  // Check if circuit size is above threshold
188  if (circ.num_qubits < qubit_threshold) {
189  data.add_metadata("fusion", metadata);
190  return;
191  }
192  // Apply fusion
193  bool applied = false;
194 
195  uint_t fusion_start = 0;
196  for (uint_t op_idx = 0; op_idx < circ.ops.size(); ++op_idx) {
197  if (can_ignore(circ.ops[op_idx]))
198  continue;
199  if (!can_apply_fusion(circ.ops[op_idx], max_fused_qubits, method)) {
200  applied |= fusion_start != op_idx && aggregate_operations(
201  circ.ops, fusion_start, op_idx, max_fused_qubits, data, method);
202  fusion_start = op_idx + 1;
203  }
204  }
205 
206  if (fusion_start < circ.ops.size() &&
207  aggregate_operations(circ.ops, fusion_start, circ.ops.size(), max_fused_qubits, data, method))
208  applied = true;
209 
210  if (applied) {
211  size_t idx = 0;
212  for (size_t i = 0; i < circ.ops.size(); ++i) {
213  if (circ.ops[i].type != optype_t::nop) {
214  if (i != idx)
215  circ.ops[idx] = circ.ops[i];
216  ++idx;
217  }
218  }
219 
220  if (idx != circ.ops.size())
221  circ.ops.erase(circ.ops.begin() + idx, circ.ops.end());
222  metadata["applied"] = true;
223 
224  // Update circuit params for fused circuit
225  circ.set_params();
226  }
227 
228  // Final metadata
229  if (verbose && applied) {
230  metadata["input_ops"] = circ.ops;
231  metadata["output_ops"] = circ.ops;
232  }
233  auto timer_stop = clock_t::now();
234  metadata["time_taken"] = std::chrono::duration<double>(timer_stop - timer_start).count();
235  data.add_metadata("fusion", metadata);
236 }

249 bool Fusion::can_apply_fusion(const op_t& op, uint_t max_fused_qubits, Method method) const {
250  if (op.conditional)
251  return false;
252  switch (op.type) {
253  case optype_t::matrix:
254  return op.mats.size() == 1 && op.qubits.size() <= max_fused_qubits;
255  case optype_t::kraus:
256  case optype_t::reset:
257  case optype_t::superop: {
258  // TODO: Add support for Method::kraus
259  return method == Method::superop && op.qubits.size() <= max_fused_qubits;
260  }
261  case optype_t::gate: {
262  if (op.qubits.size() > max_fused_qubits)
263  return false;
264  return (method == Method::unitary)
265  ? QubitUnitary::StateOpSet.contains_gates(op.name)
266  : QubitSuperoperator::StateOpSet.contains_gates(op.name);
267  }
268  case optype_t::measure:
269  case optype_t::bfunc:
270  case optype_t::roerror:
271  case optype_t::snapshot:
272  case optype_t::barrier:
273  default:
274  return false;
275  }

286 op_t Fusion::generate_fusion_operation(const std::vector<op_t>& fusioned_ops,
287  const reg_t &qubits,
288  Method method) const {
289  // Run simulation
290  RngEngine dummy_rng;
291  ExperimentData dummy_data;
292 
293  if (method == Method::unitary) {
294  // Unitary simulation
295  QubitUnitary::State<> unitary_simulator;
296  unitary_simulator.initialize_qreg(qubits.size());
297  unitary_simulator.apply_ops(fusioned_ops, dummy_data, dummy_rng);
298  auto unitary = unitary_simulator.qreg().matrix();
299  return Operations::make_unitary(qubits, std::move(unitary), std::string("fusion"));
300  }
301  // For both Kraus and SuperOp method we simulate using superoperator
302  // simulator
303  QubitSuperoperator::State<> superop_simulator;
304  superop_simulator.initialize_qreg(qubits.size());
305  superop_simulator.apply_ops(fusioned_ops, dummy_data, dummy_rng);
306  auto superop = superop_simulator.qreg().matrix();
307  if (method == Method::superop) {
308  return Operations::make_superop(qubits, std::move(superop));
309  }
310 
311  // TODO: Add support for Kraus method
312  // For Kraus method we must convert superoperator to a Kraus op
313  throw std::runtime_error("Kraus fusion method is not supported yet");
314 }

317 bool Fusion::aggregate_operations(oplist_t& ops,
318  const int fusion_start,
319  const int fusion_end,
320  uint_t max_fused_qubits,
321  ExperimentData &data,
322  Method method) const {
323 
324  // costs[i]: estimated cost to execute from 0-th to i-th in original.ops
325  std::vector<double> costs;
326  // fusion_to[i]: best path to i-th in original.ops
327  std::vector<int> fusion_to;
328 
329  // set costs and fusion_to of fusion_start
330  fusion_to.push_back(fusion_start);
331  costs.push_back(get_cost(ops[fusion_start]));
332 
333  bool applied = false;
334  // calculate the minimal path to each operation in the circuit
335  for (int i = fusion_start + 1; i < fusion_end; ++i) {
336  // init with fusion from i-th to i-th
337  fusion_to.push_back(i);
338  costs.push_back(costs[i - fusion_start - 1] + get_cost(ops[i]));
339 
340  for (int num_fusion = 2; num_fusion <= static_cast<int> (max_fused_qubits); ++num_fusion) {
341  // calculate cost if {num_fusion}-qubit fusion is applied
342  reg_t fusion_qubits;
343  add_fusion_qubits(fusion_qubits, ops[i]);
344 
345  for (int j = i - 1; j >= fusion_start; --j) {
346  add_fusion_qubits(fusion_qubits, ops[j]);
347 
348  if (static_cast<int> (fusion_qubits.size()) > num_fusion) // exceed the limit of fusion
349  break;
350 
351  // calculate a new cost of (i-th) by adding
352  double estimated_cost = estimate_cost(ops, (uint_t) j, i) // fusion gate from j-th to i-th, and
353  + (j == 0 ? 0.0 : costs[j - 1 - fusion_start]); // cost of (j-1)-th
354 
355  // update cost
356  if (estimated_cost <= costs[i - fusion_start]) {
357  costs[i - fusion_start] = estimated_cost;
358  fusion_to[i - fusion_start] = j;
359  applied = true;
360  }
361  }
362  }
363  }
364 
365  if (!applied)
366  return false;
367 
368  // generate a new circuit with the minimal path to the last operation in the circuit
369  for (int i = fusion_end - 1; i >= fusion_start;) {
370 
371  int to = fusion_to[i - fusion_start];
372 
373  if (to != i) {
374  std::vector<op_t> fusioned_ops;
375  std::set<uint_t> fusioned_qubits;
376  for (int j = to; j <= i; ++j) {
377  fusioned_ops.push_back(ops[j]);
378  fusioned_qubits.insert(ops[j].qubits.cbegin(), ops[j].qubits.cend());
379  ops[j].type = optype_t::nop;
380  }
381  if (!fusioned_ops.empty()) {
382  // We need to remap qubits in fusion subcircuits for simulation
383  // TODO: This could be done above during the fusion cost calculation
384  reg_t qubits(fusioned_qubits.begin(), fusioned_qubits.end());
385  std::unordered_map<uint_t, uint_t> qubit_mapping;
386  for (size_t j = 0; j < qubits.size(); j++) {
387  qubit_mapping[qubits[j]] = j;
388  }
389  // Remap qubits and determine method
390  bool non_unitary = false;
391  for (auto & op: fusioned_ops) {
392  non_unitary |= noise_opset_.contains(op.type);
393  for (size_t j = 0; j < op.qubits.size(); j++) {
394  op.qubits[j] = qubit_mapping[op.qubits[j]];
395  }
396  }
397  Method required_method = (non_unitary) ? method : Method::unitary;
398  ops[i] = generate_fusion_operation(fusioned_ops, qubits, required_method);
399  }
400  }
401  i = to - 1;
402  }
403  return true;
404 }

410 bool Fusion::is_diagonal(const std::vector<op_t>& ops,
411  const uint_t from,
412  const uint_t until) const {
413 
414  // check unitary matrix of ops between "from" and "to" is a diagonal matrix
415 
416  for (uint_t i = from; i <= until; ++i) {
417  // ┌───┐┌────┐┌───┐
418  // ─┤ X ├┤ U1 ├┤ X ├
419  // └─┬─┘└────┘└─┬─┘
420  // ───■──────────■─-
421  if ((i + 2) <= until
422  && ops[i + 0].name == "cx"
423  && ops[i + 1].name == "u1"
424  && ops[i + 2].name == "cx"
425  && ops[i + 0].qubits[1] == ops[i + 1].qubits[0]
426  && ops[i + 1].qubits[0] == ops[i + 2].qubits[1]
427  && ops[i + 0].qubits[0] == ops[i + 2].qubits[0] )
428  {
429  i += 2;
430  continue;
431  }
432  if (ops[i].name == "u1" || ops[i].name == "cu1")
433  continue;
434  return false;
435  }
436  return true;
437 }

439 double Fusion::estimate_cost(const std::vector<op_t>& ops,
440  const uint_t from,
441  const uint_t until) const {
442  if (is_diagonal(ops, from, until))
443  return cost_factor;
444 
445  reg_t fusion_qubits;
446  for (uint_t i = from; i <= until; ++i)
447  add_fusion_qubits(fusion_qubits, ops[i]);
448  return pow(cost_factor, (double) std::max(fusion_qubits.size() - 1, size_t(1)));
