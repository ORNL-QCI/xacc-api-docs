55 void DelayMeasure::optimize_circuit(Circuit& circ,
56  Noise::NoiseModel& noise,
57  const Operations::OpSet &allowed_opset,
58  ExperimentData &data) const {
59  // If there are no measure instructions or circuit already has
60  // no instructions after measurement, or there is quantum noise
61  // we don't need to optimize
62  if (active_ == false || circ.can_sample ||
63  !circ.opset().contains(Operations::OpType::measure) ||
64  noise.has_quantum_errors())
65  return;
66 
67  // Get position of first measurement
68  size_t pos = circ.first_measure_pos;
69 
70  // Store measure and non measure instructions in the tail
71  // for possible later remapping
72  std::vector<Operations::Op> meas_ops;
73  meas_ops.reserve(circ.ops.size() - pos);
74  std::vector<Operations::Op> non_meas_ops;
75  non_meas_ops.reserve(circ.ops.size() - pos);
76 
77  // Scan circuit to find position of first measure for all qubits;
78  std::unordered_set<uint_t> meas_qubits;
79  auto it = circ.ops.begin() + pos;
80  while (it != circ.ops.end()) {
81  // If any tail operations are conditional we abort
82  if (it->conditional)
83  return;
84  const auto type = it->type;
85  const auto qubits = it->qubits;
86  switch (type) {
87  case Operations::OpType::measure:
88  case Operations::OpType::roerror: {
89  meas_qubits.insert(qubits.begin(), qubits.end());
90  meas_ops.push_back(*it);
91  break;
92  }
93  case Operations::OpType::snapshot: {
94  return;
95  }
96  default: {
97  for (const auto& qubit : qubits) {
98  if (meas_qubits.find(qubit) != meas_qubits.end()) {
99  return;
100  }
101  }
102  non_meas_ops.push_back(*it);
103  }
104  }
105  ++it;
106  }
107  // Check if there are any non-meas instructions in the tail
108  if (non_meas_ops.empty())
109  return;
110 
111  // Now modify the circuit to add meas instructions
112  // after non meas instructions
113  circ.ops.erase(circ.ops.begin() + pos, circ.ops.end());
114  circ.ops.insert(circ.ops.end(), non_meas_ops.begin(), non_meas_ops.end());
115 
116  // Update position of first measurement
117  circ.first_measure_pos = circ.ops.size();
118  circ.can_sample = true;
119  circ.ops.insert(circ.ops.end(), meas_ops.begin(), meas_ops.end());
120 
121  if (verbose_)
122  data.add_metadata("delay_measure_verbose", circ.ops);
123 }
