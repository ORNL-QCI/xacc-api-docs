272 inline Op make_multiplexer(const reg_t &qubits,
273  const std::vector<cmatrix_t> &mats,
274  std::string label = "") {
275 
276  // Check matrices are N-qubit
277  auto dim = mats[0].GetRows();
278  auto num_targets = static_cast<uint_t>(std::log2(dim));
279  if (1ULL << num_targets != dim) {
280  throw std::invalid_argument("invalid multiplexer matrix dimension.");
281  }
282  // Check number of matrix compents is power of 2.
283  size_t num_mats = mats.size();
284  auto num_controls = static_cast<uint_t>(std::log2(num_mats));
285  if (1ULL << num_controls != num_mats) {
286  throw std::invalid_argument("invalid number of multiplexer matrices.");
287  }
288  // Check number of targets and controls matches qubits
289  if (num_controls + num_targets != qubits.size()) {
290  throw std::invalid_argument("multiplexer qubits don't match parameters.");
291  }
292  // Check each matrix component is unitary and same size
293  for (const auto &mat : mats) {
294  if (!Utils::is_unitary(mat, 1e-7))
295  throw std::invalid_argument("multiplexer matrix is not unitary.");
296  if (mat.GetRows() != dim) {
297  throw std::invalid_argument("multiplexer matrices are different size.");
298  }
299  }
300  // Get lists of controls and targets
301  reg_t controls(num_controls), targets(num_targets);
302  std::copy_n(qubits.begin(), num_controls, controls.begin());
303  std::copy_n(qubits.begin() + num_controls, num_targets, targets.begin());
304 
305  // Construct the Op
306  Op op;
307  op.type = OpType::multiplexer;
308  op.name = "multiplexer";
309  op.qubits = qubits;
310  op.mats = mats;
311  op.regs = std::vector<reg_t>({controls, targets});
312  if (label != "")
313  op.string_params = {label};
314 
315  // Validate qubits are unique.
316  check_empty_qubits(op);
317  check_duplicate_qubits(op);
318 
319  return op;
320 }

451 void add_condtional(const Allowed allowed, Op& op, const json_t &js) {
452  // Check conditional
453  if (JSON::check_key("conditional", js)) {
454  // If instruction isn't allow to be conditional throw an exception
455  if (allowed == Allowed::No) {
456  throw std::invalid_argument("Invalid instruction: \"" + op.name + "\" cannot be conditional.");
457  }
458  // If instruction is allowed to be conditional add parameters
459  if (js["conditional"].is_number()) {
460  // New style conditional
461  op.conditional_reg = js["conditional"];
462  op.conditional = true;
463  } else {
464  // DEPRECATED: old style conditional (remove in 0.3)
465  JSON::get_value(op.old_conditional_mask, "mask", js["conditional"]);
466  JSON::get_value(op.old_conditional_val, "val", js["conditional"]);
467  op.old_conditional = true;
468  }
469  }

579 Op json_to_op_bfunc(const json_t &js) {
580  Op op;
581  op.type = OpType::bfunc;
582  op.name = "bfunc";
583  op.string_params.resize(2);
584  std::string relation;
585  JSON::get_value(op.string_params[0], "mask", js); // mask hexadecimal string
586  JSON::get_value(op.string_params[1], "val", js); // value hexadecimal string
587  JSON::get_value(relation, "relation", js); // relation string
588  // Load single register / memory bit for storing result
589  uint_t tmp;
590  if (JSON::get_value(tmp, "register", js)) {
591  op.registers.push_back(tmp);
592  }
593  if (JSON::get_value(tmp, "memory", js)) {
594  op.memory.push_back(tmp);
595  }
596 
597  // Format hex strings
598  Utils::format_hex_inplace(op.string_params[0]);
599  Utils::format_hex_inplace(op.string_params[1]);
600 
601  const stringmap_t<RegComparison> comp_table({
602  {"==", RegComparison::Equal},
603  {"!=", RegComparison::NotEqual},
604  {"<", RegComparison::Less},
605  {"<=", RegComparison::LessEqual},
606  {">", RegComparison::Greater},
607  {">=", RegComparison::GreaterEqual},
608  });
609 
610  auto it = comp_table.find(relation);
611  if (it == comp_table.end()) {
612  std::stringstream msg;
613  msg << "Invalid bfunc relation string :\"" << it->first << "\"." << std::endl;
614  throw std::invalid_argument(msg.str());
615  } else {
616  op.bfunc = it->second;
617  }
618 
619  // Conditional
620  add_condtional(Allowed::No, op, js);
621 
622  // Validation
623  if (op.registers.empty()) {
624  throw std::invalid_argument("Invalid measure operation: \"register\" is empty.");
625  }
626  return op;
627 }

678  JSON::get_value(op.qubits, "qubits", js);

798 Op json_to_op_snapshot_pauli(const json_t &js) {
799  // Load default snapshot parameters
800  Op op = json_to_op_snapshot_default(js);
801 
802  // Check qubits are valid
803  check_empty_qubits(op);
804  check_duplicate_qubits(op);
805 
806  // Parse Pauli operator components
807  const auto threshold = 1e-10; // drop small components
808  // Get components
809  if (JSON::check_key("params", js) && js["params"].is_array()) {
810  for (const auto &comp : js["params"]) {
811  // Check component is length-2 array
812  if (!comp.is_array() || comp.size() != 2)
813  throw std::invalid_argument("Invalid Pauli expval snapshot (param component " +
814  comp.dump() + " invalid).");
815  // Get complex coefficient
816  complex_t coeff = comp[0];
817  // If coefficient is above threshold, get the Pauli operator string
818  // This string may contain I, X, Y, Z
819  // qubits are stored as a list where position is qubit number:
820  // eq op.qubits = [a, b, c], a is qubit-0, b is qubit-1, c is qubit-2
821  // Pauli string labels are stored in little-endian ordering:
822  // eg label = "CBA", A is the Pauli for qubit-0, B for qubit-1, C for qubit-2
823  if (std::abs(coeff) > threshold) {
824  std::string pauli = comp[1];
825  if (pauli.size() != op.qubits.size()) {
826  throw std::invalid_argument(std::string("Invalid Pauli expectation value snapshot ") +
827  "(Pauli label does not match qubit number.).");
828  }
829  // make tuple and add to components
830  op.params_expval_pauli.emplace_back(coeff, pauli);
831  } // end if > threshold
832  } // end component loop
833  } else {
834  throw std::invalid_argument("Invalid Pauli snapshot \"params\".");
835  }
836  return op;
837 }

844  const auto threshold = 1e-10; // drop small components
845  // Get matrix operator components

849  // Check component is length-2 array

853  }

860  comp.dump() + " invalid).");

864  throw std::invalid_argument("Invalid matrix expval snapshot (param component " +
865  comp.dump() + " invalid).");
866  }

871  std::unordered_set<uint_t> unique = {comp_qubits.begin(), comp_qubits.end()};

874  comp.dump() + " invalid).");
