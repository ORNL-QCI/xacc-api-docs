53  // Recursive formula to compute d = -H * g
54  // H0 = gamma0 * I is the initial approximation to H
55  // Algorithm 7.4 of Nocedal, J., & Wright, S. (2006). Numerical optimization.
56  inline void compute_direction(int k, const Scalar gamma0)
57  {
58  // L-BFGS two-loop recursion
59 
60  // Loop 1
61  m_drt.noalias() = -m_grad;
62  int bound = std::min(m_param.m, k);
63  int j = k % m_param.m;
64  for(int i = 0; i < bound; i++)
65  {
66  j = (j + m_param.m - 1) % m_param.m;
67  m_alpha[j] = m_s.col(j).dot(m_drt) / m_ys[j];
68  m_drt.noalias() -= m_alpha[j] * m_y.col(j);
69  }
70 
71  // Apply initial H0
72  m_drt *= gamma0;
73 
74  // Loop 2
75  for(int i = 0; i < bound; i++)
76  {
77  const Scalar beta = m_y.col(j).dot(m_drt) / m_ys[j];
78  m_drt.noalias() += (m_alpha[j] - beta) * m_s.col(j);
79  j = (j + 1) % m_param.m;
80  }
81  }

109  template <typename Foo>
110  inline int minimize(Foo& f, Vector& x, Scalar& fx)
111  {
112  // Dimension of the vector
113  const int n = x.size();
114  reset(n);
115 
116  // The length of lag for objective function value to test convergence
117  const int fpast = m_param.past;
118 
119  // Evaluate function and compute gradient
120  fx = f(x, m_grad);
121  Scalar xnorm = x.norm();
122  Scalar gnorm = m_grad.norm();
123  if(fpast > 0)
124  m_fx[0] = fx;
125 
126  // Early exit if the initial x is already a minimizer
127  if(gnorm <= m_param.epsilon * std::max(xnorm, Scalar(1.0)))
128  {
129  return 1;
130  }
131 
132  // Initial direction
133  m_drt.noalias() = -m_grad;
134  // Initial step size
135  Scalar step = Scalar(1.0) / m_drt.norm();
136 
137  // Number of iterations used
138  int k = 1;
139  for( ; ; )
140  {
141  // Points to the location of history that is going to be overwritten
142  // In chronological order: end, end+1, end+2, ..., m-1, 0, 1, ..., end-1
143  int end = (k - 1) % m_param.m;
144 
145  // Save the curent x and gradient
146  m_xp.noalias() = x;
147  m_gradp.noalias() = m_grad;
148 
149  // Line search to update x, fx and gradient
150  LineSearch<Scalar>::LineSearch(f, fx, x, m_grad, step, m_drt, m_xp, m_param);
151 
152  // New x norm and gradient norm
153  xnorm = x.norm();
154  gnorm = m_grad.norm();
155 
156  // Convergence test -- gradient
157  if(gnorm <= m_param.epsilon * std::max(xnorm, Scalar(1.0)))
158  {
159  return k;
160  }
161  // Convergence test -- objective function value
162  if(fpast > 0)
163  {
164  if(k >= fpast && std::abs((m_fx[k % fpast] - fx) / fx) < m_param.delta)
165  return k;
166 
167  m_fx[k % fpast] = fx;
168  }
169  // Maximum number of iterations
170  if(m_param.max_iterations != 0 && k >= m_param.max_iterations)
171  {
172  return k;
173  }
174 
175  // Update s and y
176  // s_{k+1} = x_{k+1} - x_k
177  // y_{k+1} = g_{k+1} - g_k
178  MapVec svec(&m_s(0, end), n);
179  MapVec yvec(&m_y(0, end), n);
180  svec.noalias() = x - m_xp;
181  yvec.noalias() = m_grad - m_gradp;
182 
183  // ys = y's = 1/rho
184  // yy = y'y
185  // gamma0 = ys / yy
186  const Scalar ys = yvec.dot(svec);
187  const Scalar yy = yvec.squaredNorm();
188  m_ys[end] = ys;
189  const Scalar gamma0 = ys / yy;
190 
191  // Recursive formula to compute d = -H * g
192  compute_direction(k, gamma0);
193 
194  // Reset step = 1.0 as initial guess for the next line search
195  step = Scalar(1.0);
196  k++;
197  }
198 
199  return k;
200  }
