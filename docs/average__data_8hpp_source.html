96 template <typename T>
97 T AverageData<T>::variance() const {
98  // If no counts zero or we haven't been computing variance
99  // we return default value
100  if (count_ == 0 || variance_ == false) {
101  return T();
102  }
103 
104  // If counts is 1 variance is zero
105  if (count_ == 1) {
106  return Linalg::mul(accum_squared_, 0);
107  }
108 
109  // var = (count / (count - 1)) * (accum_squared ** 2 / count - mean ** 2) for
110  // n > 1
111  T mean_squared = mean();
112  Linalg::isquare(mean_squared); // mean_squared
113  T var = Linalg::div(accum_squared_, double(count_)); // squared mean
114  Linalg::isub(var, mean_squared); // squared mean - mean squared
115  // Apply sample bias correction of counts / (counts - 1)
116  Linalg::idiv(var, double(count_) / double(count_ - 1));
117  return var;
118 }

120 template <typename T>
121 void AverageData<T>::combine(const AverageData<T> &other) {
122  // If empty we copy data without accumulating
123  if (empty()) {
124  count_ = other.count_;
125  accum_ = other.accum_;
126  variance_ = other.has_variance();
127  if (variance_) {
128  accum_squared_ = other.accum_squared_;
129  }
130  } else {
131  // Otherwise we accumulate
132  count_ += other.count_;
133  Linalg::iadd(accum_, other.accum_);
134  // If either container doesn't have variance we disable variance
135  // of the current container
136  variance_ &= other.has_variance();
137  if (variance_) {
138  Linalg::iadd(accum_squared_, other.accum_squared_);
139  }
140  }

178 void AverageData<T>::add_data(const T &datum, bool compute_variance) {
179  // If we add a single datum without variance we
180  // disable variance for the container
181  variance_ &= compute_variance;
182 
183  // For initial datum we set the accumulators to the data
184  if (count_ == 0) {
185  accum_ = datum;
186  if (has_variance()) {
187  accum_squared_ = Linalg::square(accum_);
188  }
189  } else {
