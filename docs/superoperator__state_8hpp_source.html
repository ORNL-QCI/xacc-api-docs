207 template <class data_t>
208 void State<data_t>::apply_ops(const std::vector<Operations::Op> &ops,
209  ExperimentData &data,
210  RngEngine &rng) {
211  // Simple loop over vector of input operations
212  for (const auto op: ops) {
213  switch (op.type) {
214  case Operations::OpType::barrier:
215  break;
216  case Operations::OpType::gate:
217  // Note conditionals will always fail since no classical registers
218  if (BaseState::creg_.check_conditional(op))
219  apply_gate(op);
220  break;
221  case Operations::OpType::reset:
222  apply_reset(op.qubits);
223  break;
224  case Operations::OpType::matrix:
225  apply_matrix(op.qubits, op.mats[0]);
226  break;
227  case Operations::OpType::diagonal_matrix:
228  BaseState::qreg_.apply_diagonal_matrix(op.qubits, op.params);
229  break;
230  case Operations::OpType::kraus:
231  apply_kraus(op.qubits, op.mats);
232  break;
233  case Operations::OpType::superop:
234  BaseState::qreg_.apply_superop_matrix(op.qubits, Utils::vectorize_matrix(op.mats[0]));
235  break;
236  case Operations::OpType::snapshot:
237  apply_snapshot(op, data);
238  break;
239  default:
240  throw std::invalid_argument("QubitSuperoperator::State::invalid instruction \'" +
241  op.name + "\'.");
242  }
243  }

247 size_t State<data_t>::required_memory_mb(uint_t num_qubits,
248  const std::vector<Operations::Op> &ops)
249  const {
250  // An n-qubit unitary as 2^4n complex doubles
251  // where each complex double is 16 bytes
252  (void)ops; // avoid unused variable compiler warning
253  size_t shift_mb = std::max<int_t>(0, num_qubits + 4 - 20);
254  size_t mem_mb = 1ULL << (4 * shift_mb);
255  return mem_mb;

335  const std::vector<cmatrix_t> &kmats) {
336  // Convert to Superoperator
337  const auto nrows = kmats[0].GetRows();
338  cmatrix_t superop(nrows * nrows, nrows * nrows);
339  for (const auto kraus : kmats) {
340  superop += Utils::tensor_product(Utils::conjugate(kraus), kraus);
341  }
342  BaseState::qreg_.apply_superop_matrix(qubits, Utils::vectorize_matrix(superop));

349 template <class data_t>
350 void State<data_t>::apply_gate(const Operations::Op &op) {
351  // Look for gate name in gateset
352  auto it = gateset_.find(op.name);
353  if (it == gateset_.end())
354  throw std::invalid_argument("Unitary::State::invalid gate instruction \'" +
355  op.name + "\'.");
356  switch (it -> second) {
357  case Gates::u3:
358  apply_gate_u3(op.qubits[0],
359  std::real(op.params[0]),
360  std::real(op.params[1]),
361  std::real(op.params[2]));
362  break;
363  case Gates::u2:
364  apply_gate_u3(op.qubits[0],
365  M_PI / 2.,
366  std::real(op.params[0]),
367  std::real(op.params[1]));
368  break;
369  case Gates::u1:
370  apply_gate_phase(op.qubits[0], std::exp(complex_t(0., 1.) * op.params[0]));
371  break;
372  case Gates::cx:
373  BaseState::qreg_.apply_cnot(op.qubits[0], op.qubits[1]);
374  break;
375  case Gates::cz:
376  BaseState::qreg_.apply_cz(op.qubits[0], op.qubits[1]);
377  break;
378  case Gates::id:
379  break;
380  case Gates::x:
381  BaseState::qreg_.apply_x(op.qubits[0]);
382  break;
383  case Gates::y:
384  BaseState::qreg_.apply_y(op.qubits[0]);
385  break;
386  case Gates::z:
387  BaseState::qreg_.apply_z(op.qubits[0]);
388  break;
389  case Gates::h:
390  apply_gate_u3(op.qubits[0], M_PI / 2., 0., M_PI);
391  break;
392  case Gates::s:
393  apply_gate_phase(op.qubits[0], complex_t(0., 1.));
394  break;
395  case Gates::sdg:
396  apply_gate_phase(op.qubits[0], complex_t(0., -1.));
397  break;
398  case Gates::t: {
399  const double isqrt2{1. / std::sqrt(2)};
400  apply_gate_phase(op.qubits[0], complex_t(isqrt2, isqrt2));
401  } break;
402  case Gates::tdg: {
403  const double isqrt2{1. / std::sqrt(2)};
404  apply_gate_phase(op.qubits[0], complex_t(isqrt2, -isqrt2));
405  } break;
406  case Gates::swap: {
407  BaseState::qreg_.apply_swap(op.qubits[0], op.qubits[1]);
408  } break;
409  case Gates::ccx:
410  BaseState::qreg_.apply_toffoli(op.qubits[0], op.qubits[1], op.qubits[2]);
411  break;
412  default:
413  // We shouldn't reach here unless there is a bug in gateset
414  throw std::invalid_argument("Superoperator::State::invalid gate instruction \'" +
415  op.name + "\'.");
416  }
417 }
