61 Qobj::Qobj(const json_t &js) {
62  // Check required fields
63  if (JSON::get_value(id, "qobj_id", js) == false) {
64  throw std::invalid_argument(R"(Invalid qobj: no "qobj_id" field)");
65  };
66  JSON::get_value(type, "type", js);
67  if (type != "QASM") {
68  throw std::invalid_argument(R"(Invalid qobj: "type" != "QASM".)");
69  };
70  if (JSON::check_key("experiments", js) == false) {
71  throw std::invalid_argument(R"(Invalid qobj: no "experiments" field.)");
72  }
73 
74  // Get header and config;
75  JSON::get_value(config, "config", js);
76  JSON::get_value(header, "header", js);
77 
78  // Check for fixed simulator seed
79  // If simulator seed is set, each experiment will be set to a fixed (but different) seed
80  // Otherwise a random seed will be chosen for each experiment
81  int_t seed = -1;
82  uint_t seed_shift = 0;
83  bool has_simulator_seed = JSON::get_value(seed, "seed_simulator", config);
84  const json_t &circs = js["experiments"];
85  const size_t num_circs = circs.size();
86 
87  // Check if parameterized qobj
88  // It should be of the form
89  // [exp0_params, exp1_params, ...]
90  // where:
91  // expk_params = [((i, j), pars), ....]
92  // i is the instruction index in the experiment
93  // j is the param index in the instruction
94  // pars = [par0, par1, ...] is a list of different parameterizations
95  using pos_t = std::pair<uint_t, uint_t>;
96  using exp_params_t = std::vector<std::pair<pos_t, std::vector<double>>>;
97  std::vector<exp_params_t> param_table;
98  JSON::get_value(param_table, "parameterizations", config);
99 
100  // Validate parameterizations for number of circuis
101  if (!param_table.empty() && param_table.size() != num_circs) {
102  throw std::invalid_argument(
103  R"(Invalid parameterized qobj: "parameterizations" length does not match number of circuits.)");
104  }
105 
106  // Load circuits
107  for (size_t i=0; i<num_circs; i++) {
108  // Get base circuit from qobj
109  Circuit circuit(circs[i], config);
110  if (param_table.empty() || param_table[i].empty()) {
111  // Non parameterized circuit
112  circuits.push_back(circuit);
113  } else {
114  // Load different parameterizations of the initial circuit
115  const auto circ_params = param_table[i];
116  const size_t num_params = circ_params[0].second.size();
117  const size_t num_instr = circuit.ops.size();
118  for (size_t j=0; j<num_params; j++) {
119  // Make a copy of the initial circuit
120  Circuit param_circuit = circuit;
121  for (const auto &params : circ_params) {
122  const auto instr_pos = params.first.first;
123  const auto param_pos = params.first.second;
124  // Validation
125  if (instr_pos >= num_instr) {
126  throw std::invalid_argument(R"(Invalid parameterized qobj: instruction position out of range)");
127  }
128  auto &op = param_circuit.ops[instr_pos];
129  if (param_pos >= op.params.size()) {
130  throw std::invalid_argument(R"(Invalid parameterized qobj: instruction param position out of range)");
131  }
132  if (j >= params.second.size()) {
133  throw std::invalid_argument(R"(Invalid parameterized qobj: parameterization value out of range)");
134  }
135  // Update the param
136  op.params[param_pos] = params.second[j];
137  }
138  circuits.push_back(param_circuit);
139  }
140  }
141  }
142  // Override random seed with fixed seed if set
143  // We shift the seed for each successive experiment
144  // So that results aren't correlated between experiments
145  if (has_simulator_seed) {
146  for (auto& circuit : circuits) {
147  circuit.seed = seed + seed_shift;
148  seed_shift += 2113; // Shift the seed
149  }
150  }
151 }
