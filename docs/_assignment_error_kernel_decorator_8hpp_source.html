32  std::vector<std::string> generatePermutations(int num_bits) {
33  int pow_bits = std::pow(2, num_bits);
34  std::vector<std::string> bitstring(pow_bits);
35  std::string str = "";
36  std::string curr;
37  int counter = 0;
38  int j = num_bits;
39  while (j--) {
40  str.push_back('0');
41  }
42  for (int k = 0; k <= num_bits; k++) {
43  str[num_bits - k] = '1';
44  curr = str;
45  do {
46  bitstring[counter] = curr;
47  counter++;
48  } while (next_permutation(curr.begin(), curr.end()));
49  }
50  return bitstring;
51  }

54  void generateKernel(std::shared_ptr<AcceleratorBuffer> buffer) {
55  // pow(2, num_bits) gets used a lot, so I figured I should just define it
56  // here
57 
58  int num_bits = buffer->size();
59  int pow_bits = std::pow(2, num_bits);
60  this->permutations = generatePermutations(num_bits);
61 
62  // permutations contains all possible permutationss to generate circuits,
63  // there is direct mapping from permutations to circuit as follows:
64  // permutations: 10 => X gate on zeroth qubit and nothing on first qubit and
65  // measure both permutations: 11 => X gate on both qubits and measure both
66  // etc...
67 
68  std::shared_ptr<AcceleratorBuffer> tmpBuffer = xacc::qalloc(buffer->size());
69  // generating list of circuits to evaluate
70  std::vector<std::shared_ptr<CompositeInstruction>> circuits;
71  auto provider = xacc::getIRProvider("quantum");
72  for (int i = 0; i < pow_bits; i++) {
73  auto circuit = provider->createComposite(permutations[i]);
74  int j = num_bits-1;
75  for (char c : permutations[i]) {
76  if (c == '1') {
77  // std::cout<<"1 found at position: "<<j<<std::endl;
78  //std::cout<<"j = "<<j<<std::endl;
79  auto x = provider->createInstruction("X", j);
80  circuit->addInstruction(x);
81  }
82  j = (j-1)%num_bits;
83  }
84  for(int i = 0; i < num_bits; i++){
85  circuit->addInstruction(provider->createInstruction("Measure", i));
86  }
87  if (!layout.empty()){
88  circuit->mapBits(layout);
89  }
90  //std::cout<<circuit->toString()<<std::endl;
91  //decoratedAccelerator->execute(tmpBuffer, circuit);
92  circuits.push_back(circuit);
93  }
94  //std::cout<<tmpBuffer->size()<<std::endl;
95  decoratedAccelerator->execute(tmpBuffer, circuits);
96  auto buffers = tmpBuffer->getChildren();
97  int shots = 0;
98  // compute num shots;
99  //std::cout<<tmpBuffer->nChildren()<<std::endl;
100  for (auto &x : buffers[0]->getMeasurementCounts()) {
101  shots += x.second;
102  }
103 
104  //std::cout<<"num_shots = " << shots <<std::endl;
105  // initializing vector of vector of counts to size 2^num_bits x 2^num_bits
106  std::vector<std::vector<int>> counts(std::pow(2, num_bits),
107  std::vector<int>(pow_bits));
108 
109  int row = 0;
110  Eigen::MatrixXd K(pow_bits, pow_bits);
111  K.setZero();
112  for (auto &b : buffers) {
113  //b->print();
114  int col = 0;
115  for (auto &x : permutations) {
116  auto temp = b->computeMeasurementProbability(x);
117  K(row, col) = temp;
118  col++;
119  }
120  row++;
121  }
122  //std::cout << "MATRIX:\n" << K << "\n";
123  errorKernel = K.inverse();
124  //std::cout << "INVERSE:\n" << errorKernel << "\n";
125  std::vector<double> vec(errorKernel.data(), errorKernel.data() + errorKernel.rows()*errorKernel.cols());
126  buffer->addExtraInfo("error-kernel", vec);
127 
128  gen_kernel = false;
129  }
