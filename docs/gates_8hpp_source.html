109 U1Sample::U1Sample(double lambda)
110 {
111  // Shift parameter into +- 2 Pi
112  uint_t shift_factor = std::floor(std::abs(lambda)/(2*M_PI));
113  if (shift_factor != 0)
114  {
115  if(lambda < 0)
116  {
117  lambda += shift_factor*(2*M_PI);
118  }
119  else
120  {
121  lambda -= shift_factor*(2*M_PI);
122  }
123  }
124  // Shift parameter into +- Pi
125  if (lambda > M_PI)
126  {
127  lambda -= 2*M_PI;
128  }
129  else if (lambda < -1*M_PI)
130  {
131  lambda += 2*M_PI;
132  }
133  // Compute the coefficients
134  double angle = std::abs(lambda);
135  bool s_z_quadrant = (angle > M_PI/2);
136  if(s_z_quadrant)
137  {
138  angle = angle - M_PI/2;
139  }
140  angle /= 2;
141  complex_t coeff_0 = std::cos(angle)-std::sin(angle);
142  complex_t coeff_1 = root2*std::sin(angle);
143  if(lambda < 0)
144  {
145  coeff_0 *= root_omega_star;
146  coeff_1 = coeff_1 * root_omega;
147  if(s_z_quadrant)
148  {
149  branches =
150  {
151  sample_branch_t(coeff_0, Gates::sdg),
152  sample_branch_t(coeff_1, Gates::z)
153  };
154  }
155  else
156  {
157  branches =
158  {
159  sample_branch_t(coeff_0, Gates::id),
160  sample_branch_t(coeff_1, Gates::sdg)
161  };
162  }
163  }
164  else
165  {
166  coeff_0 *= root_omega;
167  coeff_1 = coeff_1 * root_omega_star;
168  if(s_z_quadrant)
169  {
170  branches =
171  {
172  sample_branch_t(coeff_0, Gates::s),
173  sample_branch_t(coeff_1, Gates::z)
174  };
175  }
176  else
177  {
178  branches =
179  {
180  sample_branch_t(coeff_0, Gates::id),
181  sample_branch_t(coeff_1, Gates::s)
182  };
183  }
184  }
185  p_threshold = std::abs(coeff_0) / (std::abs(coeff_0)+std::abs(coeff_1));
186 }
