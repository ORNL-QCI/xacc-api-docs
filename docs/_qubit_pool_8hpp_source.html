43  // generate the pool
44  std::vector<std::shared_ptr<Observable>> generate(const int &nQubits) override {
45 
46  // The qubit pool vanishes for strings with an even number of Pauli Y's
47  // and can have at most 4 operators
48  // We loop over the indices for qubits q0-q3 and
49  // {X, Y, X} for each qubit and make sure that we have the appropriate
50  // number of Y's and only add unique operators, i.e., those which
51  // survive symmetry operations in the PauliOperator class.
52 
53  std::vector<PauliOperator> pauliOps;
54  std::vector<std::string> label;
55  std::string opStr;
56  PauliOperator op;
57 
58  for (int q0 = 0; q0 < nQubits; q0++){
59 
60  for (auto op0 : {"X", "Y", "Z"}){
61 
62  label.push_back(op0);
63  if (op0 == "Y"){
64  opStr = "Y" + std::to_string(q0);
65  op.fromString("(0,1)" + opStr);
66  pauliOps.push_back(op);
67  }
68 
69  for (int q1 = q0 + 1; q1 < nQubits; q1++){
70 
71  for (auto op1 : {"X", "Y", "Z"}){
72 
73  label.push_back(op1);
74  if (std::count(label.begin(), label.end(), "Y") == 1){
75  opStr = op0 + std::to_string(q0) + op1 + std::to_string(q1);
76  op.fromString("(0,1)" + opStr);
77  if(!(std::find(pauliOps.begin(), pauliOps.end(), op) != pauliOps.end())){
78  pauliOps.push_back(op);
79  }
80 
81  }
82 
83  for (int q2 = q1 + 1; q2 < nQubits; q2++){
84 
85  for (auto op2 : {"X", "Y", "Z"}){
86 
87  label.push_back(op2);
88  if (std::count(label.begin(), label.end(), "Y") == 1 || std::count(label.begin(), label.end(), "Y") == 3){
89 
90  opStr = op0 + std::to_string(q0) + op1 + std::to_string(q1) + op2 + std::to_string(q2);
91  op.fromString("(0,1)" + opStr);
92  if(!(std::find(pauliOps.begin(), pauliOps.end(), op) != pauliOps.end())){
93  pauliOps.push_back(op);
94  }
95  }
96 
97  for (int q3 = q2 + 1; q3 < nQubits; q3++){
98 
99  for (auto op3 : {"X", "Y", "Z"}){
100 
101  label.push_back(op3);
102  if (std::count(label.begin(), label.end(), "Y") == 1 || std::count(label.begin(), label.end(), "Y") == 3){
103 
104  opStr = op0 + std::to_string(q0) + op1 + std::to_string(q1) + op2 + std::to_string(q2) + op3 + std::to_string(q3);
105  op.fromString("(0,1)" + opStr);
106  if(!(std::find(pauliOps.begin(), pauliOps.end(), op) != pauliOps.end())){
107  pauliOps.push_back(op);
108  }
109 
110  }
111 
112  label.pop_back();
113  }
114  }
115  label.pop_back();
116  }
117  }
118  label.pop_back();
119  }
120  }
121  label.pop_back();
122  }
123  }
124 
125  for (auto op: pauliOps){
126  pool.push_back(std::dynamic_pointer_cast<Observable>(std::make_shared<PauliOperator>(op)));
127  }
128 
129  return pool;
130  }

148  std::make_pair("param_id", "x" + std::to_string(varIdx)),
149  std::make_pair("no-i", true)});
150 
151  return gate;
152 
153  }
