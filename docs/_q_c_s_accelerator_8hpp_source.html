108  msgpack::unpacked request(T &requestType, zmq::socket_t &socket) {
109  msgpack::sbuffer sbuf;
110  msgpack::pack(sbuf, requestType);

123  void getSocketURLs() {
124  if (!xacc::directoryExists(std::getenv("HOME") + std::string("/.qcs"))) {
125  xacc::error("[QCS] $HOME/.qcs directory does not exists.");
126  }
127 
128  if (!xacc::fileExists(std::getenv("HOME") + std::string("/.qcs_config"))) {
129  xacc::error("[QCS] $HOME/.qcs_config file does not exist.");
130  }
131 
132  if (!xacc::fileExists(std::getenv("HOME") +
133  std::string("/.qcs/user_auth_token"))) {
134  xacc::error("[QCS] $HOME/.qcs/user_auth_token file does not exist.");
135  }
136  std::ifstream stream(std::getenv("HOME") +
137  std::string("/.qcs/user_auth_token"));
138  std::string contents((std::istreambuf_iterator<char>(stream)),
139  std::istreambuf_iterator<char>());
140 
141  // std::cout << "HELLO2:\n" << contents << "\n";
142  auto auth_json = json::parse(contents);
143 
144  std::string graph_ql_query =
145  "\\n mutation Engage($name: String!) {\\n "
146  "engage(input: { lattice: { name: $name }}) {\\n "
147  "success\\n message\\n engagement {\\n "
148  " type\\n qpu {\\n "
149  "endpoint\\n credentials {\\n "
150  " clientPublic\\n clientSecret\\n "
151  " serverPublic\\n }\\n }\\n "
152  " compiler {\\n endpoint\\n "
153  " }\\n expiresAt\\n }\\n "
154  "}\\n }\\n ";
155  std::string json_data = "{\"query\": \"" + graph_ql_query +
156  "\", \"variables\": {\"name\": \"" + backend +
157  "\"}}";
158  std::map<std::string, std::string> headers{
159  {"Content-Type", "application/json"},
160  {"Connection", "keep-alive"},
161  {"Accept", "application/octet-stream"},
162  {"Content-Length", std::to_string(json_data.length())},
163  {"Authorization",
164  "Bearer " + auth_json["access_token"].get<std::string>()}};
165  auth_token = auth_json["access_token"].get<std::string>();
166 
167  // std::cout << json_data.length() << "\n";
168  // std::cout << json_data << "\n";
169  auto resp = this->post("https://dispatch.services.qcs.rigetti.com",
170  "/graphql", json_data, headers);
171 
172  // std::cout << "HELLO:\n" << resp << "\n";
173 
174  // this came back
175  auto resp_json = json::parse(resp);
176  qpu_endpoint = resp_json["data"]["engage"]["engagement"]["qpu"]["endpoint"]
177  .get<std::string>();
178  auto qpu_creds =
179  resp_json["data"]["engage"]["engagement"]["qpu"]["credentials"];
180  qpu_compiler_endpoint =
181  resp_json["data"]["engage"]["engagement"]["compiler"]["endpoint"]
182  .get<std::string>();
183 
184  client_public = qpu_creds["clientPublic"].get<std::string>();
185  client_secret = qpu_creds["clientSecret"].get<std::string>();
186  server_public = qpu_creds["serverPublic"].get<std::string>();
187 
188  if (qpu_compiler_endpoint.empty() || qpu_endpoint.empty()) {
189  xacc::error("QCS Error: Cannot find qpu_compiler or qpu endpoint.");
190  }

192  std::ifstream stream2(std::getenv("HOME") + std::string("/.qcs_config"));
193  std::string contents2((std::istreambuf_iterator<char>(stream2)),
194  std::istreambuf_iterator<char>());
195  auto lines = xacc::split(contents2, '\n');

202  }
203  }

216  void initialize(const HeterogeneousMap &params = {}) override {
217  if (params.stringExists("backend")) {
218  backend = params.getString("backend");
219 
220  if (backend.find("-qvm") != std::string::npos) {
221  backend.erase(backend.find("-qvm"), 4);
222  }
223 
224  Client client;
225  auto response = client.get("https://forest-server.qcs.rigetti.com",
226  "/lattices/" + backend);
227 
228  latticeJson.Parse(response);
229  const auto &oneq = latticeJson["lattice"]["isa"]["1Q"];
230  auto &twoq = latticeJson["lattice"]["isa"]["2Q"];
231 
232  for (auto itr = oneq.MemberBegin(); itr != oneq.MemberEnd(); ++itr) {
233  physicalQubits.push_back(std::stoi(itr->name.GetString()));
234  }
235  for (auto itr = twoq.MemberBegin(); itr != twoq.MemberEnd(); ++itr) {
236  auto connStr = itr->name.GetString();
237  auto split = xacc::split(connStr, '-');
238  latticeEdges.push_back({std::stoi(split[0]), std::stoi(split[1])});
239  }
240  }
