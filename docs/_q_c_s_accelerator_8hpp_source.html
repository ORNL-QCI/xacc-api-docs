74  msgpack::unpacked request(T &requestType, zmq::socket_t &socket) {
75  msgpack::sbuffer sbuf;
76  msgpack::pack(sbuf, requestType);
77  zmq::message_t msg(sbuf.size());
78  memcpy(msg.data(), sbuf.data(), sbuf.size());
79  socket.send(msg);

81  socket.recv(&reply, 0);
82  msgpack::unpacked unpackedData;
83  msgpack::unpack(unpackedData, static_cast<const char *>(reply.data()),

88  void getSocketURLs() {
89  std::ifstream stream(std::getenv("HOME") + std::string("/.forest_config"));
90  std::string contents((std::istreambuf_iterator<char>(stream)),
91  std::istreambuf_iterator<char>());
92 
93  std::vector<std::string> lines;
94  lines = xacc::split(contents, '\n');
95  for (auto l : lines) {
96  if (l.find("qpu_compiler_address") != std::string::npos) {
97  std::vector<std::string> split = xacc::split(l, '=');
98  auto key = split[1];
99  xacc::trim(key);
100  qpu_compiler_endpoint = key;
101  } else if (l.find("qpu_endpoint_address") != std::string::npos) {
102  std::vector<std::string> split;
103  split = xacc::split(l, '=');
104  auto key = split[1];
105  xacc::trim(key);
106  qpu_endpoint = key;
107  }
108  }
109 
110  if (qpu_compiler_endpoint.empty() || qpu_endpoint.empty()) {
111  xacc::error("QCS Error: Cannot find qpu_compiler or qpu endpoint.");
112  }
113  }

124  void initialize(const HeterogeneousMap &params = {}) override {
125  if (params.stringExists("qcs-backend")) {
126  backend = params.getString("qcs-backend");
127 
128  if (backend.find("-qvm") != std::string::npos) {
129  backend.erase(backend.find("-qvm"), 4);
130  }
131 
132  Client client;
133  auto response = client.get("https://forest-server.qcs.rigetti.com",
134  "/lattices/" + backend);
135 
136  latticeJson.Parse(response);
137  const auto &oneq = latticeJson["lattice"]["isa"]["1Q"];
138  auto &twoq = latticeJson["lattice"]["isa"]["2Q"];
139 
140  for (auto itr = oneq.MemberBegin(); itr != oneq.MemberEnd(); ++itr) {
141  physicalQubits.push_back(std::stoi(itr->name.GetString()));
142  }
143  for (auto itr = twoq.MemberBegin(); itr != twoq.MemberEnd(); ++itr) {
144  auto connStr = itr->name.GetString();
145  auto split = xacc::split(connStr, '-');
146  latticeEdges.push_back({std::stoi(split[0]), std::stoi(split[1])});
147  }
148  }
149 
150  // Read .forest_config
151  getSocketURLs();
152 
153  updateConfiguration(params);
154 
155  auto irt = std::make_shared<MapToPhysical>(latticeEdges);
156  xacc::contributeService("qcs-map-qubits", irt);
157  }
