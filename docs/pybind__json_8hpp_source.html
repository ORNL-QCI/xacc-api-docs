199 json_t JSON::numpy_to_json_1d(py::array_t<T, py::array::c_style> arr) {
200  py::buffer_info buf = arr.request();
201  if (buf.ndim != 1) {
202  throw std::runtime_error("Number of dims must be 1");
203  }
204 
205  T *ptr = (T *) buf.ptr;
206  size_t D0 = buf.shape[0];
207 
208  std::vector<T> tbr; // to be returned
209  for (size_t n0 = 0; n0 < D0; n0++)
210  tbr.push_back(ptr[n0]);
211 
212  return std::move(tbr);
213 }

215 template <typename T>
216 json_t JSON::numpy_to_json_2d(py::array_t<T, py::array::c_style> arr) {
217  py::buffer_info buf = arr.request();
218  if (buf.ndim != 2) {
219  throw std::runtime_error("Number of dims must be 2");
220  }
221 
222  T *ptr = (T *) buf.ptr;
223  size_t D0 = buf.shape[0];
224  size_t D1 = buf.shape[1];
225 
226  std::vector<std::vector<T > > tbr; // to be returned
227  for (size_t n0 = 0; n0 < D0; n0++) {
228  std::vector<T> tbr1;
229  for (size_t n1 = 0; n1 < D1; n1++) {
230  tbr1.push_back(ptr[n1 + D1*n0]);
231  }
232  tbr.push_back(tbr1);
233  }
234 
235  return std::move(tbr);

239 template <typename T>
240 json_t JSON::numpy_to_json_3d(py::array_t<T, py::array::c_style> arr) {
241  py::buffer_info buf = arr.request();
242  if (buf.ndim != 3) {
243  throw std::runtime_error("Number of dims must be 3");
244  }
245  T *ptr = (T *) buf.ptr;
246  size_t D0 = buf.shape[0];
247  size_t D1 = buf.shape[1];
248  size_t D2 = buf.shape[2];
249 
250  // to be returned
251  std::vector<std::vector<std::vector<T > > > tbr;
252  for (size_t n0 = 0; n0 < D0; n0++) {
253  std::vector<std::vector<T> > tbr1;
254  for (size_t n1 = 0; n1 < D1; n1++) {
255  std::vector<T> tbr2;
256  for (size_t n2 = 0; n2 < D2; n2++) {
257  tbr2.push_back(ptr[n2 + D2*(n1 + D1*n0)]);
258  }
259  tbr1.push_back(tbr2);
260  }
261  tbr.push_back(tbr1);
262  }
263 
264  return std::move(tbr);

286 void std::to_json(json_t &js, const py::handle &obj) {
287  if (py::isinstance<py::float_>(obj)) {
288  js = obj.cast<nl::json::number_float_t>();
289  } else if (py::isinstance<py::bool_>(obj)) {
290  js = obj.cast<nl::json::boolean_t>();
291  } else if (py::isinstance<py::int_>(obj)) {
292  js = obj.cast<nl::json::number_integer_t>();
293  } else if (py::isinstance<py::str>(obj)) {
294  js = obj.cast<nl::json::string_t>();
295  } else if (py::isinstance<py::tuple>(obj) || py::isinstance<py::list>(obj)) {
296  js = nl::json::array();
297  for (py::handle value: obj) {
298  js.push_back(value);
299  }
300  } else if (py::isinstance<py::dict>(obj)) {
301  for (auto item : py::cast<py::dict>(obj)) {
302  js[item.first.cast<nl::json::string_t>()] = item.second;
303  }
304  } else if (py::isinstance<py::array_t<double> >(obj)) {
305  js = JSON::numpy_to_json(obj.cast<py::array_t<double, py::array::c_style> >());
306  } else if (py::isinstance<py::array_t<std::complex<double> > >(obj)) {
307  js = JSON::numpy_to_json(obj.cast<py::array_t<std::complex<double>, py::array::c_style> >());
308  } else if (obj.is_none()) {
309  return;
310  } else {
311  auto type_str = std::string(py::str(obj.get_type()));
312  if ( type_str == "<class \'complex\'>"
313  || type_str == "<class \'numpy.complex64\'>"
314  || type_str == "<class \'numpy.complex128\'>"
315  || type_str == "<class \'numpy.complex_\'>" ) {
316  auto tmp = obj.cast<std::complex<double>>();
317  js.push_back(tmp.real());
318  js.push_back(tmp.imag());
319  } else if ( type_str == "<class \'numpy.uint32\'>"
320  || type_str == "<class \'numpy.uint64\'>"
321  || type_str == "<class \'numpy.int32\'>"
322  || type_str == "<class \'numpy.int64\'>" ) {
323  js = obj.cast<nl::json::number_integer_t>();
324  } else if ( type_str == "<class \'numpy.float32\'>"
325  || type_str == "<class \'numpy.float64\'>" ) {
326  js = obj.cast<nl::json::number_float_t>();
327  } else {
328  throw std::runtime_error("to_json not implemented for this type of object: " + std::string(py::str(obj.get_type())));
329  }
330  }

333 void std::from_json(const json_t &js, py::object &o) {
334  if (js.is_boolean()) {
335  o = py::bool_(js.get<nl::json::boolean_t>());
336  } else if (js.is_number()) {
337  if (js.is_number_float()) {
338  o = py::float_(js.get<nl::json::number_float_t>());
339  } else if (js.is_number_unsigned()) {
340  o = py::int_(js.get<nl::json::number_unsigned_t>());
341  } else {
342  o = py::int_(js.get<nl::json::number_integer_t>());
343  }
344  } else if (js.is_string()) {
345  o = py::str(js.get<nl::json::string_t>());
346  } else if (js.is_array()) {
347  std::vector<py::object> obj(js.size());
348  for (auto i = 0; i < js.size(); i++)
349  {
350  py::object tmp;
351  from_json(js[i], tmp);
352  obj[i] = tmp;
353  }
354  o = py::cast(obj);
355  } else if (js.is_object()) {
356  py::dict obj;
357  for (json_t::const_iterator it = js.cbegin(); it != js.cend(); ++it)
358  {
359  py::object tmp;
360  from_json(it.value(), tmp);
361  obj[py::str(it.key())] = tmp;
362  }
363  o = std::move(obj);
364  } else if (js.is_null()) {
365  o = py::none();
366  } else {
367  throw std::runtime_error("from_json not implemented for this json::type: " + js.dump());
368  }
369 }

380  auto tbr = py::array_t<T, py::array::f_style>(shape);
381  auto buf = tbr.template mutable_unchecked<1>();
382  for (size_t i = 0; i < src.size(); i++) {
383  buf(i) = src[i];
384  }
385  return std::move(tbr);

388 template<typename T>
389 py::object AerToPy::array_from_matrix(const matrix<T> &src) {
390  std::array<py::ssize_t, 2> shape { static_cast<py::ssize_t>(src.GetRows()), static_cast<py::ssize_t>(src.GetColumns()) };
391  auto tbr = py::array_t<T, py::array::f_style>(shape);
392  auto buf = tbr.template mutable_unchecked<2>();
393  for (size_t r = 0; r < src.GetRows(); r++) {
394  for (size_t c = 0; c < src.GetColumns(); c++)
395  buf(r, c) = src(r, c);
396  }
397  return std::move(tbr);
