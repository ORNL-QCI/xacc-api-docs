179  if (BaseVector::data_size_ == statevec.size()) {

209 template <typename data_t>
210 cvector_t<double> DensityMatrix<data_t>::vmat2vsuperop(const cvector_t<double> &vmat) const {
211  // Get dimension of unvectorized matrix
212  size_t dim = size_t(std::sqrt(vmat.size()));
213  cvector_t<double> ret(dim * dim * dim * dim, 0.);
214  for (size_t i=0; i < dim; i++)
215  for (size_t j=0; j < dim; j++)
216  for (size_t k=0; k < dim; k++)
217  for (size_t l=0; l < dim; l++)
218  ret[dim*i+k+(dim*dim)*(dim*j+l)] = std::conj(vmat[i+dim*j])*vmat[k+dim*l];
219  return ret;
220 }

234 template <typename data_t>
235 void DensityMatrix<data_t>::apply_unitary_matrix(const reg_t &qubits,
236  const cvector_t<double> &mat) {
237  // Check if we apply as two N-qubit matrix multiplications or a single 2N-qubit matrix mult.
238  if (qubits.size() > apply_unitary_threshold_) {
239  // Apply as two N-qubit matrix mults
240  auto nq = num_qubits();
241  reg_t conj_qubits;
242  for (const auto q: qubits) {
243  conj_qubits.push_back(q + nq);
244  }
245  // Apply id \otimes U
246  BaseVector::apply_matrix(qubits, mat);
247  // Apply conj(U) \otimes id
248  BaseVector::apply_matrix(conj_qubits, AER::Utils::conjugate(mat));
249  } else {
250  // Apply as single 2N-qubit matrix mult.
251  apply_superop_matrix(qubits, vmat2vsuperop(mat));
252  }
253 }

369 template <typename data_t>
370 double DensityMatrix<data_t>::expval_pauli(const reg_t &qubits,
371  const std::string &pauli) const {
372 
373  // Break string up into Z and X
374  // With Y being both Z and X (plus a phase)
375  const size_t N = qubits.size();
376  uint_t x_mask = 0;
377  uint_t z_mask = 0;
378  uint_t num_y = 0;
379  for (size_t i = 0; i < N; ++i) {
380  const auto bit = BITS[qubits[i]];
381  switch (pauli[N - 1 - i]) {
382  case 'I':
383  break;
384  case 'X': {
385  x_mask += bit;
386  break;
387  }
388  case 'Z': {
389  z_mask += bit;
390  break;
391  }
392  case 'Y': {
393  x_mask += bit;
394  z_mask += bit;
395  num_y++;
396  break;
397  }
398  default:
399  throw std::invalid_argument("Invalid Pauli \"" + std::to_string(pauli[N - 1 - i]) + "\".");
400  }
401  }
402 
403  // Special case for only I Paulis
404  if (x_mask + z_mask == 0) {
405  return std::real(BaseMatrix::trace());
406  }
407 
408  // Compute the overall phase of the operator.
409  // This is (-1j) ** number of Y terms modulo 4
410  std::complex<data_t> phase(1, 0);
411  switch (num_y & 3) {
412  case 0:
413  // phase = 1
414  break;
415  case 1:
416  // phase = -1j
417  phase = std::complex<data_t>(0, -1);
418  break;
419  case 2:
420  // phase = -1
421  phase = std::complex<data_t>(-1, 0);
422  break;
423  case 3:
424  // phase = 1j
425  phase = std::complex<data_t>(0, 1);
426  break;
427  }
428  // The shift for density matrix indices in the vectorized vector
429  const size_t start = 0;
430  const size_t stop = BITS[num_qubits()];
431  auto lambda = [&](const int_t i, double &val_re, double &val_im)->void {
432  (void)val_im; // unused
433  auto val = std::real(phase * BaseVector::data_[i ^ x_mask + stop * i]);
434  if (z_mask) {
435  // Portable implementation of __builtin_popcountll
436  auto count = i & z_mask;
437  count = (count & 0x5555555555555555) + ((count >> 1) & 0x5555555555555555);
438  count = (count & 0x3333333333333333) + ((count >> 2) & 0x3333333333333333);
439  count = (count & 0x0f0f0f0f0f0f0f0f) + ((count >> 4) & 0x0f0f0f0f0f0f0f0f);
440  count = (count & 0x00ff00ff00ff00ff) + ((count >> 8) & 0x00ff00ff00ff00ff);
441  count = (count & 0x0000ffff0000ffff) + ((count >> 16) & 0x0000ffff0000ffff);
442  count = (count & 0x00000000ffffffff) + ((count >> 32) & 0x00000000ffffffff);
443  if (count & 1) {
444  val = -val;
445  }
446  }
447  val_re += val;
448  };
449  return std::real(BaseVector::apply_reduction_lambda(lambda, start, stop));
450 }
