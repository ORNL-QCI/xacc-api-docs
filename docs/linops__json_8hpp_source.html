31 json_t& iadd(json_t& lhs, const json_t& rhs) {
32  // Null case
33  if (lhs.is_null()) {
34  lhs = rhs;
35  return lhs;
36  }
37  if (rhs.is_null()) {
38  return lhs;
39  }
40  // Terminating case
41  if (lhs.is_number() && rhs.is_number()) {
42  lhs = double(lhs) + double(rhs);
43  return lhs;
44  }
45  // Recursive cases
46  if (lhs.is_array() && rhs.is_array() && lhs.size() == rhs.size()) {
47  for (size_t pos = 0; pos < lhs.size(); pos++) {
48  iadd(lhs[pos], rhs[pos]);
49  }
50  } else if (lhs.is_object() && rhs.is_object()) {
51  for (auto it = rhs.begin(); it != rhs.end(); ++it) {
52  iadd(lhs[it.key()], it.value());
53  }
54  } else {
55  throw std::invalid_argument("Input JSONs cannot be added.");
56  }
57  return lhs;
58 }
