108 void Circuit::set_params() {
109 
110  // Clear current containers
111  opset_ = Operations::OpSet();
112  qubitset_.clear();
113  memoryset_.clear();
114  registerset_.clear();
115  can_sample = true;
116  first_measure_pos = 0;
117 
118  // Check maximum qubit, and register size
119  // Memory size is loaded from qobj config
120  // Also check if measure sampling is in principle possible
121  bool first_measure = true;
122  for (size_t i = 0; i < ops.size(); ++i) {
123  const auto& op = ops[i];
124  has_conditional |= op.conditional;
125  opset_.insert(op);
126  qubitset_.insert(op.qubits.begin(), op.qubits.end());
127  memoryset_.insert(op.memory.begin(), op.memory.end());
128  registerset_.insert(op.registers.begin(), op.registers.end());
129 
130  // Compute measure sampling check
131  if (can_sample) {
132  if (first_measure) {
133  if (op.type == OpType::measure || op.type == OpType::roerror) {
134  first_measure = false;
135  } else {
136  first_measure_pos++;
137  }
138  } else if(!(op.type == OpType::barrier ||
139  op.type == OpType::measure ||
140  op.type == OpType::roerror)) {
141  can_sample = false;
142  }
143  }
144  }
145 
146  // Get required number of qubits, memory, registers from set maximums
147  // Since these are std::set containers the largest element is the
148  // Last element of the (non-empty) container.
149  num_qubits = (qubitset_.empty()) ? 0 : 1 + *qubitset_.rbegin();
150  num_memory = (memoryset_.empty()) ? 0 : 1 + *memoryset_.rbegin();
151  num_registers = (registerset_.empty()) ? 0 : 1 + *registerset_.rbegin();
152 }

161 Circuit::Circuit(const json_t &circ, const json_t &qobj_config) : Circuit() {
162 
163  // Get config
164  json_t config = qobj_config;
165  if (JSON::check_key("config", circ)) {
166  for (auto it = circ["config"].cbegin(); it != circ["config"].cend();
167  ++it) {
168  config[it.key()] = it.value(); // overwrite circuit level config values
169  }
170  }
171  // Load instructions
172  if (JSON::check_key("instructions", circ) == false) {
173  throw std::invalid_argument("Invalid Qobj experiment: no \"instructions\" field.");
174  }
175  ops.clear(); // remove any current operations
176  const json_t &jops = circ["instructions"];
177  for(auto jop: jops){
178  ops.emplace_back(Operations::json_to_op(jop));
179  }
180 
181  // Set circuit parameters from ops
182  set_params();
183 
184  // Load metadata
185  JSON::get_value(header, "header", circ);
186  JSON::get_value(shots, "shots", config);
187 
188  // Check for specified memory slots
189  uint_t memory_slots = 0;
190  JSON::get_value(memory_slots, "memory_slots", config);
191  if (memory_slots < num_memory) {
192  throw std::invalid_argument("Invalid Qobj experiment: not enough memory slots.");
193  }
194  // override memory slot number
195  num_memory = memory_slots;
196 
197  // Check for specified n_qubits
198  if (JSON::check_key("n_qubits", config)) {
199  uint_t n_qubits = config["n_qubits"];
200  if (n_qubits < num_qubits) {
201  throw std::invalid_argument("Invalid Qobj experiment: n_qubits < instruction qubits.");
202  }
203  // override qubit number
204  num_qubits = n_qubits;
205  }
206 }
