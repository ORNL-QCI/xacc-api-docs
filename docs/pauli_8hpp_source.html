58 Pauli::Pauli(const std::string &label) {
59  const auto num_qubits = label.size();
60  X = BV::BinaryVector(num_qubits);
61  Z = BV::BinaryVector(num_qubits);
62  // The label corresponds to tensor product order
63  // So the first element of label is the last qubit and vice versa
64  for (size_t i =0; i < num_qubits; i++) {
65  const auto qubit_i = num_qubits - 1 - i;
66  switch (label[i]) {
67  case 'I':
68  break;
69  case 'X':
70  X.set1(qubit_i);
71  break;
72  case 'Y':
73  X.set1(qubit_i);
74  Z.set1(qubit_i);
75  break;
76  case 'Z':
77  Z.set1(qubit_i);
78  break;
79  default:
80  throw std::invalid_argument("Invalid Pauli label");
81  }
82  }
83 }

85 std::string Pauli::str() const {
86  // Check X and Z are same length
87  const auto num_qubits = X.getLength();
88  if (Z.getLength() != num_qubits)
89  throw std::runtime_error("Pauli::str X and Z vectors are different length.");
90  std::string label;
91  for (size_t i =0; i < num_qubits; i++) {
92  const auto qubit_i = num_qubits - 1 - i;
93  if (!X[qubit_i] && !Z[qubit_i])
94  label.push_back('I');
95  else if (X[qubit_i] && !Z[qubit_i])
96  label.push_back('X');
97  else if (X[qubit_i] && Z[qubit_i])
98  label.push_back('Y');
99  else if (!X[qubit_i] && Z[qubit_i])
100  label.push_back('Z');
101  }
102  return label;
103 }

105 int8_t Pauli::phase_exponent(const Pauli& pauli1, const Pauli& pauli2) {
106  int8_t exponent = 0;
107  for (size_t q = 0; q < pauli1.X.getLength(); q++) {
108  exponent += pauli2.X[q] * pauli1.Z[q] * (1 + 2 * pauli2.Z[q] + 2 * pauli1.X[q]);
109  exponent -= pauli1.X[q] * pauli2.Z[q] * (1 + 2 * pauli1.Z[q] + 2 * pauli2.X[q]);
110  exponent %= 4;
111  }
112  if (exponent < 0)
113  exponent += 4;
114  return exponent;
115 }
