116 void ClassicalRegister::store_measure(const reg_t &outcome,
117  const reg_t &memory,
118  const reg_t &registers) {
119  // Assumes memory and registers are either empty or same size as outcome!
120  bool use_mem = !memory.empty();
121  bool use_reg = !registers.empty();
122  for (size_t j=0; j < outcome.size(); j++) {
123  if (use_mem) {
124  // least significant bit first ordering
125  const size_t pos = creg_memory_.size() - memory[j] - 1;
126  creg_memory_[pos] = std::to_string(outcome[j])[0]; // int->string->char
127  }
128  if (use_reg) {
129  // least significant bit first ordering
130  const size_t pos = creg_register_.size() - registers[j] - 1;
131  creg_register_[pos] = std::to_string(outcome[j])[0]; // int->string->char
132  }

137 bool ClassicalRegister::check_conditional(const Operations::Op &op) const {
138  // Check if op is conditional
139  if (op.conditional)
140  return (creg_register_[creg_register_.size() - op.conditional_reg - 1] == '1');
141 
142  // DEPRECATED: old style conditional
143  if (op.old_conditional) {
144  std::string current;
145  auto mask = Utils::padleft(Utils::hex2bin(op.old_conditional_mask, false),
146  '0', creg_memory_.size());
147  for (size_t pos=0; pos < mask.size(); pos++) {
148  if (mask[pos] == '1')
149  current.push_back(creg_memory_[pos]);
150  }
151  auto val = Utils::padleft(Utils::hex2bin(op.old_conditional_val, false),
152  '0', current.size());
153  return (val == current);
154  }
155 
156  // Op is not conditional
157  return true;

161 void ClassicalRegister::apply_bfunc(const Operations::Op &op) {
162 
163  // Check input is boolean function op
164  if (op.type != Operations::OpType::bfunc) {
165  throw std::invalid_argument("ClassicalRegister::apply_bfunc: Input is not a bfunc op.");
166  }
167 
168  const std::string &mask = op.string_params[0];
169  const std::string &target_val = op.string_params[1];
170  int_t compared; // if equal this should be 0, if less than -1, if greater than +1
171 
172  // Check if register size fits into a 64-bit integer
173  if (creg_register_.size() <= 64) {
174  uint_t reg_int = std::stoull(creg_register_, nullptr, 2); // stored as bitstring
175  uint_t mask_int = std::stoull(mask, nullptr, 16); // stored as hexstring
176  uint_t target_int = std::stoull(target_val, nullptr, 16); // stored as hexstring
177  compared = (reg_int & mask_int) - target_int;
178  } else {
179  // We need to use big ints so we implement the bit-mask via the binary string
180  // representation rather than using a big integer class
181  std::string mask_bin = Utils::hex2bin(mask); // has 0b prefix while creg_register_ doesn't
182  size_t length = std::min(mask_bin.size() - 2, creg_register_.size()); // -2 to remove 0b
183  std::string masked_val = std::string(length, '0');
184  for (size_t rev_pos = 0; rev_pos < length; rev_pos++) {
185  masked_val[length - 1 - rev_pos] = (mask_bin[mask_bin.size() - 1 - rev_pos]
186  & creg_register_[creg_register_.size() - 1 - rev_pos]);
187  }
188  masked_val = Utils::bin2hex(masked_val); // convert to hex string
189  // Using string comparison to compare to target value
190  compared = masked_val.compare(target_val);
191  }
192  // check value of compared integer for different comparison operations
193  bool outcome;
194  switch (op.bfunc) {
195  case Operations::RegComparison::Equal:
196  outcome = (compared == 0);
197  break;
198  case Operations::RegComparison::NotEqual:
199  outcome = (compared != 0);
200  break;
201  case Operations::RegComparison::Less:
202  outcome = (compared < 0);
203  break;
204  case Operations::RegComparison::LessEqual:
205  outcome = (compared <= 0);
206  break;
207  case Operations::RegComparison::Greater:
208  outcome = (compared > 0);
209  break;
210  case Operations::RegComparison::GreaterEqual:
211  outcome = (compared >= 0);
212  break;
213  default:
214  // we shouldn't ever get here
215  throw std::invalid_argument("Invalid boolean function relation.");
216  }
217  // Store outcome in register
218  if (op.registers.size() > 0) {
219  const size_t pos = creg_register_.size() - op.registers[0] - 1;
220  creg_register_[pos] = (outcome) ? '1' : '0';
221  }
222  // Optionally store outcome in memory
223  if (op.memory.size() > 0) {
224  const size_t pos = creg_memory_.size() - op.memory[0] - 1;
225  creg_memory_[pos] = (outcome) ? '1' : '0';
226  }
227 }

229 // Apply readout error instruction to classical registers
230 void ClassicalRegister::apply_roerror(const Operations::Op &op, RngEngine &rng) {
231 
232  // Check input is readout error op
233  if (op.type != Operations::OpType::roerror) {
234  throw std::invalid_argument("ClassicalRegister::apply_roerror Input is not a readout error op.");
235  }
236 
237  // Get current classical bit (and optionally register bit) values
238  std::string mem_str;
239 
240  // Get values of bits as binary string
241  // We iterate from the end of the list of memory bits
242  for (auto it = op.memory.rbegin(); it < op.memory.rend(); ++it) {
243  auto bit = *it;
244  mem_str.push_back(creg_memory_[creg_memory_.size() - 1 - bit]);
245  }
246  auto mem_val = std::stoull(mem_str, nullptr, 2);
247  auto outcome = rng.rand_int(op.probs[mem_val]);
248  auto noise_str = Utils::int2string(outcome, 2, op.memory.size());
249  for (size_t pos = 0; pos < op.memory.size(); ++pos) {
250  auto bit = op.memory[pos];
251  creg_memory_[creg_memory_.size() - 1 - bit] = noise_str[noise_str.size() - 1 - pos];
252  }
253  // and the same error to register classical bits if they are used
254  for (size_t pos = 0; pos < op.registers.size(); ++pos) {
255  auto bit = op.registers[pos];
256  creg_register_[creg_register_.size() - 1 - bit] = noise_str[noise_str.size() - 1 - pos];
257  }
258 }
